     1тЖТ     1тЖТ//! # Tools тАФ OS-control tool definitions for the Aether agentic layer.
     2тЖТ     2тЖТ//!
     3тЖТ     3тЖТ//! This module defines the **tool surface** the autonomous agent can use to
     4тЖТ     4тЖТ//! perceive and manage a Linux-like OS. Each tool is:
     5тЖТ     5тЖТ//!
     6тЖТ     6тЖТ//! - a [`Tool`] enum variant (carries the parsed parameters),
     7тЖТ     7тЖТ//! - a [`ToolSpec`] entry in the [`ToolRegistry`] catalog (carries the
     8тЖТ     8тЖТ//!   name + description + JSON-Schema parameters used to advertise the tool
     9тЖТ     9тЖТ//!   to the LLM and to the `GET /v1/tools` introspection endpoint),
    10тЖТ    10тЖТ//! - a dispatch arm in [`ToolRegistry::execute`] (carries the placeholder
    11тЖТ    11тЖТ//!   executor).
    12тЖТ    12тЖТ//!
    13тЖТ    13тЖТ//! # Why placeholders?
    14тЖТ    14тЖТ//!
    15тЖТ    15тЖТ//! The Aether Engine runs as a Rust HTTP service; it does not itself touch
    16тЖТ    16тЖТ//! the filesystem, spawn shells, or open GUI windows. The actual OS
    17тЖТ    17тЖТ//! integration happens through the Next.js layer in Alpha-OS, which:
    18тЖТ    18тЖТ//!
    19тЖТ    19тЖТ//! 1. Calls `POST /v1/agent/run` to start the agentic loop,
    20тЖТ    20тЖТ//! 2. Receives the parsed [`ToolCall`](crate::agent::ToolCall)s back as JSON,
    21тЖТ    21тЖТ//! 3. Performs the real OS action (file IO, PTY exec, window manager
    22тЖТ    22тЖТ//!    mutation, graph memory add, тАж),
    23тЖТ    23тЖТ//! 4. Feeds the resulting observation back into the next agent iteration via
    24тЖТ    24тЖТ//!    the `context` field of the next `POST /v1/agent/run` call.
    25тЖТ    25тЖТ//!
    26тЖТ    26тЖТ//! The placeholder executors here return **canonical "what would happen"
    27тЖТ    27тЖТ//! responses** so the agent loop can be tested end-to-end (perceive тЖТ think
    28тЖТ    28тЖТ//! тЖТ act тЖТ observe) before the OS wiring is complete, and so the structure
    29тЖТ    29тЖТ//! of the tool registry is exercised by every iteration even when no real
    30тЖТ    30тЖТ//! side-effects are available.
    31тЖТ    31тЖТ//!
    32тЖТ    32тЖТ//! # Adding a new tool
    33тЖТ    33тЖТ//!
    34тЖТ    34тЖТ//! 1. Add a variant to [`Tool`] carrying its parsed parameters.
    35тЖТ    35тЖТ//! 2. Add a `(name, description, schema)` triple to
    36тЖТ    36тЖТ//!    [`ToolRegistry::catalog`].
    37тЖТ    37тЖТ//! 3. Add a dispatch arm to [`ToolRegistry::execute`] returning a
    38тЖТ    38тЖТ//!    placeholder result.
    39тЖТ    39тЖТ//! 4. Add a constructor arm to [`Tool::from_request`].
    40тЖТ    40тЖТ
    41тЖТ    41тЖТuse serde_json::{json, Value};

    47тЖТ    47тЖТ/// A parsed tool call тАФ the variant identifies *which* tool, and the fields
    48тЖТ    48тЖТ/// carry the validated parameters extracted from the LLM's JSON payload by
    49тЖТ    49тЖТ/// [`Tool::from_request`].
    50тЖТ    50тЖТ///
    51тЖТ    51тЖТ/// Variants are constructed only by [`Tool::from_request`]; the agent loop
    52тЖТ    52тЖТ/// never builds them directly. Unknown tool names or missing required
    53тЖТ    53тЖТ/// parameters yield `None`, which the agent loop surfaces as a
    54тЖТ    54тЖТ/// `tool_not_found` error message back to the LLM.
    55тЖТ    55тЖТ#[derive(Clone, Debug)]

    57тЖТ    57тЖТ    /// Read a file from disk. Returns the file contents as a UTF-8 string.
    58тЖТ    58тЖТ    FileRead {
    59тЖТ    59тЖТ        /// Absolute or relative path of the file to read.
    60тЖТ    60тЖТ        path: String,
    61тЖТ    61тЖТ    },
    62тЖТ    62тЖТ    /// Write `content` to `path`, overwriting if the file already exists and
    63тЖТ    63тЖТ    /// creating it (including parent directories) if it does not.
    64тЖТ    64тЖТ    FileWrite {
    65тЖТ    65тЖТ        /// Path of the file to write.
    66тЖТ    66тЖТ        path: String,
    67тЖТ    67тЖТ        /// The full text content to write.
    68тЖТ    68тЖТ        content: String,
    69тЖТ    69тЖТ    },
    70тЖТ    70тЖТ    /// List the entries of a directory. Returns `{ name, is_dir }` per entry.
    71тЖТ    71тЖТ    FileList {
    72тЖТ    72тЖТ        /// Path of the directory to list.
    73тЖТ    73тЖТ        path: String,
    74тЖТ    74тЖТ    },
    75тЖТ    75тЖТ    /// Delete a file or directory. Directories are removed recursively.
    76тЖТ    76тЖТ    FileDelete {
    77тЖТ    77тЖТ        /// Path of the file or directory to delete.
    78тЖТ    78тЖТ        path: String,
    79тЖТ    79тЖТ    },
    80тЖТ    80тЖТ    /// Run a shell command in the user's default shell. Returns
    81тЖТ    81тЖТ    /// `{ stdout, stderr, exit_code }`.
    82тЖТ    82тЖТ    Exec {
    83тЖТ    83тЖТ        /// The full shell command line (interpreted by `/bin/bash -c`).
    84тЖТ    84тЖТ        command: String,
    85тЖТ    85тЖТ    },
    86тЖТ    86тЖТ    /// Open an application window in the Alpha-OS desktop.
    87тЖТ    87тЖТ    WindowOpen {
    88тЖТ    88тЖТ        /// App identifier (e.g. `"terminal"`, `"editor"`, `"browser"`).
    89тЖТ    89тЖТ        app: String,
    90тЖТ    90тЖТ    },
    91тЖТ    91тЖТ    /// Close an open application window by its id.
    92тЖТ    92тЖТ    WindowClose {
    93тЖТ    93тЖТ        /// The window id returned by `window_open` or listed in the OS state.
    94тЖТ    94тЖТ        window_id: String,
    95тЖТ    95тЖТ    },
    96тЖТ    96тЖТ    /// Add a memory to Akasha (the semantic memory graph). The memory
    97тЖТ    97тЖТ    /// becomes a new node and is automatically linked to its semantic
    98тЖТ    98тЖТ    /// neighbors.
    99тЖТ    99тЖТ    MemoryAdd {
   100тЖТ   100тЖТ        /// The memory text.
   101тЖТ   101тЖТ        text: String,
   102тЖТ   102тЖТ        /// Memory kind: `"fact"`, `"lesson"`, `"plan"`, `"goal"`,
   103тЖТ   103тЖТ        /// `"intention"`, `"log"`, or `"code"`.
   104тЖТ   104тЖТ        kind: String,
   105тЖТ   105тЖТ    },
   106тЖТ   106тЖТ    /// Search the semantic memory graph for memories matching `query`.
   107тЖТ   107тЖТ    /// Returns the top results with cosine-similarity scores.
   108тЖТ   108тЖТ    MemorySearch {
   109тЖТ   109тЖТ        /// The free-text query to search for.
   110тЖТ   110тЖТ        query: String,
   111тЖТ   111тЖТ    },
   112тЖТ   112тЖТ    /// Search the public web for `query`. Returns a list of result snippets
   113тЖТ   113тЖТ    /// with URLs.
   114тЖТ   114тЖТ    WebSearch {
   115тЖТ   115тЖТ        /// The search query.
   116тЖТ   116тЖТ        query: String,
   117тЖТ   117тЖТ    },
   118тЖТ   118тЖТ    /// Create a multi-step plan. The plan is stored and surfaced back to the
   119тЖТ   119тЖТ    /// agent on subsequent iterations as part of the OS state, so the agent
   120тЖТ   120тЖТ    /// can track its progress through the steps.
   121тЖТ   121тЖТ    PlanCreate {
   122тЖТ   122тЖТ        /// The high-level goal the plan achieves.
   123тЖТ   123тЖТ        goal: String,
   124тЖТ   124тЖТ        /// The ordered list of step descriptions.
   125тЖТ   125тЖТ        steps: Vec<String>,
   126тЖТ   126тЖТ    },
   127тЖТ   127тЖТ    /// Mark a step in the current plan as done. Steps are 0-indexed.
   128тЖТ   128тЖТ    PlanUpdate {
   129тЖТ   129тЖТ        /// The index of the step to mark as done.
   130тЖТ   130тЖТ        step_index: usize,
   131тЖТ   131тЖТ    },
   132тЖТ   132тЖТ}

   190тЖТ   190тЖТ
   191тЖТ   191тЖТ    /// Construct a [`Tool`] from a `(name, params)` pair extracted from the
   192тЖТ   192тЖТ    /// LLM's tool-call payload by
   193тЖТ   193тЖТ    /// [`parse_tool_calls`](crate::agent::parse_tool_calls).
   194тЖТ   194тЖТ    ///
   195тЖТ   195тЖТ    /// Returns `None` when:
   196тЖТ   196тЖТ    /// - the tool name is unknown to the registry,
   197тЖТ   197тЖТ    /// - a required parameter is missing,
   198тЖТ   198тЖТ    /// - a parameter has the wrong JSON type.
   199тЖТ   199тЖТ    ///
   200тЖТ   200тЖТ    /// The agent loop converts `None` into a `tool_not_found` /
   201тЖТ   201тЖТ    /// `invalid_params` tool-result message so the LLM can self-correct on
   202тЖТ   202тЖТ    /// the next iteration.
   203тЖТ   203тЖТ    pub fn from_request(name: &str, params: &Value) -> Option<Self> {
   204тЖТ   204тЖТ        match name {
   205тЖТ   205тЖТ            "file_read" => {
   206тЖТ   206тЖТ                let path = params.get("path")?.as_str()?.to_string();
   207тЖТ   207тЖТ                Some(Tool::FileRead { path })
   208тЖТ   208тЖТ            }
   209тЖТ   209тЖТ            "file_write" => {
   210тЖТ   210тЖТ                let path = params.get("path")?.as_str()?.to_string();
   211тЖТ   211тЖТ                let content = params
   212тЖТ   212тЖТ                    .get("content")
   213тЖТ   213тЖТ                    .and_then(|c| c.as_str())
   214тЖТ   214тЖТ                    .unwrap_or("")
   215тЖТ   215тЖТ                    .to_string();
   216тЖТ   216тЖТ                Some(Tool::FileWrite { path, content })
   217тЖТ   217тЖТ            }
   218тЖТ   218тЖТ            "file_list" => {
   219тЖТ   219тЖТ                let path = params.get("path")?.as_str()?.to_string();
   220тЖТ   220тЖТ                Some(Tool::FileList { path })
   221тЖТ   221тЖТ            }
   222тЖТ   222тЖТ            "file_delete" => {
   223тЖТ   223тЖТ                let path = params.get("path")?.as_str()?.to_string();
   224тЖТ   224тЖТ                Some(Tool::FileDelete { path })
   225тЖТ   225тЖТ            }
   226тЖТ   226тЖТ            "exec" => {
   227тЖТ   227тЖТ                let command = params.get("command")?.as_str()?.to_string();
   228тЖТ   228тЖТ                Some(Tool::Exec { command })
   229тЖТ   229тЖТ            }
   230тЖТ   230тЖТ            "window_open" => {
   231тЖТ   231тЖТ                let app = params.get("app")?.as_str()?.to_string();
   232тЖТ   232тЖТ                Some(Tool::WindowOpen { app })
   233тЖТ   233тЖТ            }
   234тЖТ   234тЖТ            "window_close" => {
   235тЖТ   235тЖТ                let window_id = params.get("window_id")?.as_str()?.to_string();
   236тЖТ   236тЖТ                Some(Tool::WindowClose { window_id })
   237тЖТ   237тЖТ            }
   238тЖТ   238тЖТ            "memory_add" => {
   239тЖТ   239тЖТ                let text = params.get("text")?.as_str()?.to_string();
   240тЖТ   240тЖТ                let kind = params
   241тЖТ   241тЖТ                    .get("kind")
   242тЖТ   242тЖТ                    .and_then(|k| k.as_str())
   243тЖТ   243тЖТ                    .unwrap_or("fact")
   244тЖТ   244тЖТ                    .to_string();
   245тЖТ   245тЖТ                Some(Tool::MemoryAdd { text, kind })
   246тЖТ   246тЖТ            }
   247тЖТ   247тЖТ            "memory_search" => {
   248тЖТ   248тЖТ                let query = params.get("query")?.as_str()?.to_string();
   249тЖТ   249тЖТ                Some(Tool::MemorySearch { query })
   250тЖТ   250тЖТ            }
   251тЖТ   251тЖТ            "web_search" => {
   252тЖТ   252тЖТ                let query = params.get("query")?.as_str()?.to_string();
   253тЖТ   253тЖТ                Some(Tool::WebSearch { query })
   254тЖТ   254тЖТ            }
   255тЖТ   255тЖТ            "plan_create" => {
   256тЖТ   256тЖТ                let goal = params.get("goal")?.as_str()?.to_string();
   257тЖТ   257тЖТ                let steps = params
   258тЖТ   258тЖТ                    .get("steps")
   259тЖТ   259тЖТ                    .and_then(|s| s.as_array())
   260тЖТ   260тЖТ                    .map(|arr| {
   261тЖТ   261тЖТ                        arr.iter()
   262тЖТ   262тЖТ                            .filter_map(|s| s.as_str().map(|s| s.to_string()))
   263тЖТ   263тЖТ                            .collect::<Vec<_>>()
   264тЖТ   264тЖТ                    })
   265тЖТ   265тЖТ                    .unwrap_or_default();
   266тЖТ   266тЖТ                Some(Tool::PlanCreate { goal, steps })
   267тЖТ   267тЖТ            }
   268тЖТ   268тЖТ            "plan_update" => {
   269тЖТ   269тЖТ                let step_index = params.get("step_index")?.as_u64()? as usize;
   270тЖТ   270тЖТ                Some(Tool::PlanUpdate { step_index })
   271тЖТ   271тЖТ            }
   272тЖТ   272тЖТ            _ => None,
   273тЖТ   273тЖТ        }
   274тЖТ   274тЖТ    }
   275тЖТ   275тЖТ}

   297тЖТ   297тЖТ// ToolRegistry тАФ catalog + dispatch
   298тЖТ   298тЖТ// ---------------------------------------------------------------------------
   299тЖТ   299тЖТ
   300тЖТ   300тЖТ/// Stateless registry of all tools available to the autonomous agent.
   301тЖТ   301тЖТ///
   302тЖТ   302тЖТ/// Despite having no fields, this is a struct (not free functions) so that
   303тЖТ   303тЖТ/// future evolution (per-agent tool whitelisting, dynamic tool registration,
   304тЖТ   304тЖТ/// rate limits, capability scopes, тАж) has an obvious extension point. Today
   305тЖТ   305тЖТ/// every call to [`ToolRegistry::new`] returns an identical catalog.
   306тЖТ   306тЖТ///
   307тЖТ   307тЖТ/// # Example
   308тЖТ   308тЖТ///
   309тЖТ   309тЖТ/// ```ignore
   310тЖТ   310тЖТ/// use crate::tools::{Tool, ToolRegistry};
   311тЖТ   311тЖТ///
   312тЖТ   312тЖТ/// let registry = ToolRegistry::new();
   313тЖТ   313тЖТ/// let catalog = registry.catalog();
   314тЖТ   314тЖТ/// assert_eq!(catalog.len(), 12);
   315тЖТ   315тЖТ///
   316тЖТ   316тЖТ/// let params = serde_json::json!({ "path": "/etc/hostname" });
   317тЖТ   317тЖТ/// let tool = Tool::from_request("file_read", &params).unwrap();
   318тЖТ   318тЖТ/// let (output, success) = futures::executor::block_on(registry.execute(&tool));
   319тЖТ   319тЖТ/// assert!(success);
   320тЖТ   320тЖТ/// ```
   321тЖТ   321тЖТpub struct ToolRegistry;
   322тЖТ   322тЖТ
   323тЖТ   323тЖТimpl ToolRegistry {
   324тЖТ   324тЖТ    /// Construct a new registry. Today this is a no-op (the registry is
   325тЖТ   325тЖТ    /// stateless); the constructor exists for future extensibility.
   326тЖТ   326тЖТ    pub fn new() -> Self {
   327тЖТ   327тЖТ        Self
   328тЖТ   328тЖТ    }
   329тЖТ   329тЖТ
   330тЖТ   330тЖТ    /// Return the full catalog of tools in a stable order. The order is
   331тЖТ   331тЖТ    /// preserved across calls so the system prompt and the introspection
   332тЖТ   332тЖТ    /// endpoint emit identical tool lists.
   333тЖТ   333тЖТ    ///
   334тЖТ   334тЖТ    /// The 12 tools cover: file IO (`file_read`, `file_write`, `file_list`,
   335тЖТ   335тЖТ    /// `file_delete`), shell (`exec`), window manager (`window_open`,
   336тЖТ   336тЖТ    /// `window_close`), memory/Akasha (`memory_add`, `memory_search`),
   337тЖТ   337тЖТ    /// external information (`web_search`), and planning (`plan_create`,
   338тЖТ   338тЖТ    /// `plan_update`).
   339тЖТ   339тЖТ    pub fn catalog(&self) -> Vec<ToolSpec> {
   340тЖТ   340тЖТ        vec![
   341тЖТ   341тЖТ            ToolSpec {

   527тЖТ   527тЖТ    /// Execute a parsed [`Tool`].
   528тЖТ   528тЖТ    ///
   529тЖТ   529тЖТ    /// Returns `(output, success)` where:
   530тЖТ   530тЖТ    /// - `output` is a human/LLM-readable string describing the result,
   531тЖТ   531тЖТ    /// - `success` is `true` when the tool "succeeded" (placeholder logic;
   532тЖТ   532тЖТ    ///   see the module docs for why real OS integration happens via the
   533тЖТ   533тЖТ    ///   Next.js layer).
   534тЖТ   534тЖТ    ///
   535тЖТ   535тЖТ    /// Placeholder behavior:
   536тЖТ   536тЖТ    ///
   537тЖТ   537тЖТ    /// - File/shell/window/web tools always succeed and return a canonical
   538тЖТ   538тЖТ    ///   `[<tool> placeholder] would <action>: <params>` string. The Next.js
   539тЖТ   539тЖТ    ///   layer intercepts these and performs the real side-effect.
   540тЖТ   540тЖТ    /// - Memory tools (`memory_add`, `memory_search`) and plan tools
   541тЖТ   541тЖТ    ///   (`plan_create`, `plan_update`) also return placeholders тАФ the
   542тЖТ   542тЖТ    ///   Next.js layer is responsible for POSTing to `/graph/add` /
   543тЖТ   543тЖТ    ///   `/graph/search` and updating the plan store respectively.
   544тЖТ   544тЖТ    ///
   545тЖТ   545тЖТ    /// This split keeps the tool registry self-contained (no `AppState`
   546тЖТ   546тЖТ    /// dependency, no async I/O for files, no PTY plumbing) while still
   547тЖТ   547тЖТ    /// exercising the full agent loop end-to-end.

   549тЖТ   549тЖТ        match tool {

   552тЖТ   552тЖТ                    "[file_read placeholder] would read file at \"{path}\" and return its contents."

   555тЖТ   555тЖТ            ),
   556тЖТ   556тЖТ            Tool::FileWrite { path, content } => (
   557тЖТ   557тЖТ                format!(
   558тЖТ   558тЖТ                    "[file_write placeholder] would write {} byte(s) to \"{path}\".",
   559тЖТ   559тЖТ                    content.len()
   560тЖТ   560тЖТ                ),
   561тЖТ   561тЖТ                true,
   562тЖТ   562тЖТ            ),
   563тЖТ   563тЖТ            Tool::FileList { path } => (

   569тЖТ   569тЖТ            Tool::FileDelete { path } => (

   571тЖТ   571тЖТ                    "[file_delete placeholder] would delete the file or directory at \"{path}\"."
   572тЖТ   572тЖТ                ),
   573тЖТ   573тЖТ                true,
   574тЖТ   574тЖТ            ),
   575тЖТ   575тЖТ            Tool::Exec { command } => (

   648тЖТ   648тЖТ/// Truncate `s` to at most `max` characters, appending an ellipsis when
   649тЖТ   649тЖТ/// truncation occurs. Used to keep placeholder tool outputs short enough
   650тЖТ   650тЖТ/// to fit comfortably inside the LLM's context window across many
   651тЖТ   651тЖТ/// iterations.
   652тЖТ   652тЖТfn truncate(s: &str, max: usize) -> String {
   653тЖТ   653тЖТ    if s.chars().count() <= max {
   654тЖТ   654тЖТ        s.to_string()
   655тЖТ   655тЖТ    } else {
   656тЖТ   656тЖТ        let truncated: String = s.chars().take(max).collect();
   657тЖТ   657тЖТ        format!("{truncated}тАж")
   658тЖТ   658тЖТ    }
   659тЖТ   659тЖТ}
