     1‚Üí//! # Agent ‚Äî the autonomous OS-management agentic loop.
     2‚Üí//!
     3‚Üí//! This module is the **brain** of the autonomous OS layer. It implements
     4‚Üí//! the classical perceive ‚Üí think ‚Üí act ‚Üí observe control loop on top of
     5‚Üí//! the Aether Engine's existing inference pipeline, turning a small GGUF
     6‚Üí//! model into an autonomous agent capable of managing a Linux-like OS:
     7‚Üí//!
     8‚Üí//! 1. **Perceive** ‚Äî receive the current OS state (open windows, memory
     9‚Üí//!    snapshot, active plan, goals) as a `context` JSON object.
    10‚Üí//! 2. **Think** ‚Äî call the LLM through the engine's OpenAI-compatible
    11‚Üí//!    backend, primed with a system prompt that frames the model as an
    12‚Üí//!    autonomous OS agent and lists every available tool's schema.
    13‚Üí//! 3. **Act** ‚Äî parse the LLM's response for tool calls (see
    14‚Üí//!    [`parse_tool_calls`], which supports both bare inline JSON and
    15‚Üí//!    fenced-code-block formats).
    16‚Üí//! 4. **Execute** ‚Äî dispatch each parsed [`ToolCall`] through the
    17‚Üí//!    [`ToolRegistry`](crate::tools::ToolRegistry); on success, capture the
    18‚Üí//!    tool's output. On failure (unknown tool, invalid params, executor
    19‚Üí//!    error), capture a diagnostic instead.
    20‚Üí//! 5. **Observe** ‚Äî feed every tool result back into the message stream as
    21‚Üí//!    a new tool-role message, so the LLM sees the consequence of its
    22‚Üí//!    action on the next think step.
    23‚Üí//! 6. **Repeat** ‚Äî continue until the LLM emits the `TASK_COMPLETE` sentinel,
    24‚Üí//!    emits a response with no tool calls, or the iteration cap is hit.
    25‚Üí//!
    26‚Üí//! # Architecture
    27‚Üí//!
    28‚Üí//! ```text
    29‚Üí//!                 ‚îå‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îê
    30‚Üí//!   goal ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚ñ∂‚îÇ  run_agent_loop              ‚îÇ
    31‚Üí//!   context ‚îÄ‚îÄ‚îÄ‚îÄ‚ñ∂‚îÇ  ‚îå‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îê  ‚îÇ
    32‚Üí//!                 ‚îÇ  ‚îÇ 1. Perceive (state)    ‚îÇ  ‚îÇ
    33‚Üí//!                 ‚îÇ  ‚îÇ 2. Think  (LLM call)   ‚îÇ  ‚îÇ
    34‚Üí//!                 ‚îÇ  ‚îÇ 3. Act    (parse calls)‚îÇ  ‚îÇ
    35‚Üí//!                 ‚îÇ  ‚îÇ 4. Execute (registry)  ‚îÇ  ‚îÇ
    36‚Üí//!                 ‚îÇ  ‚îÇ 5. Observe (feed back) ‚îÇ  ‚îÇ
    37‚Üí//!                 ‚îÇ  ‚îÇ 6. Repeat / stop       ‚îÇ  ‚îÇ
    38‚Üí//!                 ‚îÇ  ‚îî‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îò  ‚îÇ
    39‚Üí//!                 ‚îÇ  ‚îÄ‚îÄ‚ñ∂ AgentRunResult           ‚îÇ
    40‚Üí//!                 ‚îî‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îò
    41‚Üí//! ```
    42‚Üí//!
    43‚Üí//! The loop is **stateless across requests**: every `POST /v1/agent/run`
    44‚Üí//! call starts a fresh [`AgentState`]. Persistence (memory, plans, goals)
    45‚Üí//! is delegated to the existing Aether Engine endpoints (`/graph/add`,
    46‚Üí//! `/graph/search`) and to the Next.js layer's plan store. The `context`
    47‚Üí//! field on the request body is the bridge ‚Äî the caller assembles the
    48‚Üí//! current OS snapshot into it before each call.
    49‚Üí//!
    50‚Üí//! # Why this lives in the inference engine
    51‚Üí//!
    52‚Üí//! Putting the agent loop in the Aether Engine (rather than the Next.js
    53‚Üí//! layer) means the loop benefits from all ten engine innovations: every
    54‚Üí//! think step's system prompt can be augmented by graph retrieval, every
    55‚Üí//! repeated goal can hit the action cache, every complex step can be
    56‚Üí//! decomposed, and every response is ATD-verified. The agent loop is a
    57‚Üí//! *client* of the cognitive pipeline, not a replacement for it.
    58‚Üí//!
    59‚Üí//! # Termination
    60‚Üí//!
    61‚Üí//! The loop stops when any of the following is true:
    62‚Üí//!
    63‚Üí//! - The LLM's response contains the literal `TASK_COMPLETE` sentinel
    64‚Üí//!   (case-sensitive). `completed` is set to `true`.
    65‚Üí//! - The LLM's response contains no parseable tool calls. `completed` is
    66‚Üí//!   `false` (the agent stopped without an explicit completion signal).
    67‚Üí//! - `max_iterations` is reached. `completed` is `false`.
    68‚Üí//!
    69‚Üí//! The caller decides what to do with an incomplete result (retry, surface
    70‚Üí//! to the user, etc.).

    81‚Üí/// A single message in the agent's conversation log.
    82‚Üí///
    83‚Üí/// Roles mirror the OpenAI chat-completions spec, with one extension:
    84‚Üí/// `tool` is used for tool-result messages so the caller can distinguish
    85‚Üí/// them from ordinary user messages in the log. When serialized for the
    86‚Üí/// backend LLM call ([`AgentMessage::to_request_json`]), `tool` messages
    87‚Üí/// are rewritten to `user` role with a `[tool_result: <tool>]` prefix,
    88‚Üí/// because small GGUF models frequently reject the `tool` role without a
    89‚Üí/// matching `tool_call_id`. The semantic content is preserved.
    90‚Üí#[derive(Clone, Debug, Serialize, Deserialize)]
    91‚Üípub struct AgentMessage {
    92‚Üí    /// `"system"`, `"user"`, `"assistant"`, or `"tool"`.
    93‚Üí    pub role: String,
    94‚Üí    /// The message text. For `tool` messages this is the tool's output.
    95‚Üí    pub content: String,
    96‚Üí    /// For `tool` messages: the name of the tool that produced this output.
    97‚Üí    /// `None` for all other roles.
    98‚Üí    #[serde(skip_serializing_if = "Option::is_none")]
    99‚Üí    pub name: Option<String>,
   100‚Üí}
   101‚Üí
   102‚Üíimpl AgentMessage {
   103‚Üí    /// Construct a `system` message.
   104‚Üí    pub fn system(content: impl Into<String>) -> Self {
   105‚Üí        Self {
   106‚Üí            role: "system".to_string(),
   107‚Üí            content: content.into(),
   108‚Üí            name: None,
   109‚Üí        }
   110‚Üí    }
   111‚Üí
   112‚Üí    /// Construct a `user` message.
   113‚Üí    pub fn user(content: impl Into<String>) -> Self {
   114‚Üí        Self {
   115‚Üí            role: "user".to_string(),
   116‚Üí            content: content.into(),
   117‚Üí            name: None,
   118‚Üí        }
   119‚Üí    }
   120‚Üí
   121‚Üí    /// Construct an `assistant` message (the LLM's prior response).
   122‚Üí    pub fn assistant(content: impl Into<String>) -> Self {
   123‚Üí        Self {
   124‚Üí            role: "assistant".to_string(),
   125‚Üí            content: content.into(),
   126‚Üí            name: None,
   127‚Üí        }
   128‚Üí    }
   129‚Üí
   130‚Üí    /// Construct a `tool` message capturing a tool's output. `tool_name`
   131‚Üí    /// is recorded in [`name`](Self::name) and used to prefix the content
   132‚Üí    /// when serializing for the backend.
   133‚Üí    pub fn tool(tool_name: impl Into<String>, content: impl Into<String>) -> Self {
   134‚Üí        Self {
   135‚Üí            role: "tool".to_string(),
   136‚Üí            content: content.into(),
   137‚Üí            name: Some(tool_name.into()),
   138‚Üí        }
   139‚Üí    }
   140‚Üí
   141‚Üí    /// Serialize this message into the OpenAI chat-completions `messages[]`
   142‚Üí    /// shape for the backend LLM call. See the type docs for the `tool` ‚Üí
   143‚Üí    /// `user` rewrite rationale.
   144‚Üí    pub fn to_request_json(&self) -> Value {
   145‚Üí        match self.role.as_str() {
   146‚Üí            "tool" => json!({
   147‚Üí                "role": "user",
   148‚Üí                "content": format!(
   149‚Üí                    "[tool_result: {}] {}",
   150‚Üí                    self.name.as_deref().unwrap_or("unknown"),
   151‚Üí                    self.content
   152‚Üí                ),
   153‚Üí            }),
   154‚Üí            other => json!({
   155‚Üí                "role": other,
   156‚Üí                "content": self.content,
   157‚Üí            }),
   158‚Üí        }
   159‚Üí    }
   160‚Üí}

   162‚Üí/// A parsed tool call extracted from the LLM's response text.

   164‚Üí/// Constructed by [`parse_tool_calls`]. The `name` field is matched

   169‚Üípub struct ToolCall {
   170‚Üí    /// Canonical tool name (e.g. `"file_read"`, `"exec"`).

   172‚Üí    /// JSON parameters object as parsed from the LLM payload.
   173‚Üí    pub params: Value,
   174‚Üí}
   175‚Üí
   176‚Üí/// The result of executing one [`ToolCall`]. Captured into
   177‚Üí/// [`AgentState::tool_results`] and fed back to the LLM as a `tool`-role

   179‚Üí#[derive(Clone, Debug, Serialize, Deserialize)]
   180‚Üípub struct ToolResult {
   181‚Üí    /// Name of the tool that produced this result (echoed back from the

   183‚Üí    pub tool: String,
   184‚Üí    /// Human/LLM-readable output string. On failure this is a diagnostic

   186‚Üí    pub output: String,
   187‚Üí    /// `true` when the tool executor returned success, `false` on any kind
   188‚Üí    /// of failure (unknown tool, bad params, executor error).
   189‚Üí    pub success: bool,
   190‚Üí}
   191‚Üí
   192‚Üí/// The LLM's response on one think step. Parsed from the raw backend
   193‚Üí/// response text by [`extract_assistant_content`]; the optional
   194‚Üí/// `tool_calls` field is filled in by the caller using
   195‚Üí/// [`parse_tool_calls`].
   196‚Üí#[derive(Clone, Debug)]
   197‚Üípub struct AgentResponse {
   198‚Üí    /// The full assistant text. May contain zero or more tool-call JSON
   199‚Üí    /// payloads plus free-form reasoning.
   200‚Üí    pub text: String,
   201‚Üí    /// Tool calls parsed from `text`. Empty when the LLM made no tool
   202‚Üí    /// calls this iteration.
   203‚Üí    pub tool_calls: Vec<ToolCall>,
   204‚Üí}
   205‚Üí
   206‚Üí/// Mutable state carried across iterations of the agent loop.
   207‚Üí///
   208‚Üí/// A fresh [`AgentState`] is constructed at the top of
   209‚Üí/// [`run_agent_loop`]; each iteration reads from it (messages ‚Üí LLM call)

   212‚Üí#[derive(Clone, Debug)]
   213‚Üípub struct AgentState {
   214‚Üí    /// 1-indexed iteration counter. `0` before the first think step.
   215‚Üí    pub iteration: usize,
   216‚Üí    /// The full message stream: `[system, user(goal), assistant, tool,

   218‚Üí    pub messages: Vec<AgentMessage>,
   219‚Üí    /// Every tool result produced so far, in execution order. Used by the
   220‚Üí    /// handler to surface a transcript of what the agent did.
   221‚Üí    pub tool_results: Vec<ToolResult>,
   222‚Üí    /// `true` only when the LLM emitted the `TASK_COMPLETE` sentinel.
   223‚Üí    pub completed: bool,
   224‚Üí}
   225‚Üí
   226‚Üí/// Final return value of [`run_agent_loop`]. Serialized by the

   243‚Üí    pub completed: bool,

   269‚Üí/// Build the system prompt that frames the LLM as an autonomous OS agent
   270‚Üí/// with access to the supplied `tool_catalog` and the supplied OS `context`.
   271‚Üí///
   272‚Üí/// The prompt is split into five sections:
   273‚Üí///
   274‚Üí/// 1. **Identity** ‚Äî "You are the autonomous agent core of Alpha-OS."
   275‚Üí/// 2. **Goal protocol** ‚Äî how to call tools (JSON object format), and how
   276‚Üí///    to signal completion (`TASK_COMPLETE`).
   277‚Üí/// 3. **Available tools** ‚Äî one block per tool with name + description +
   278‚Üí///    JSON Schema.
   279‚Üí/// 4. **Current OS state** ‚Äî the `context` JSON pretty-printed, so the
   280‚Üí///    LLM perceives the open windows, memory snapshot, active plan, etc.
   281‚Üí/// 5. **Operating principles** ‚Äî minimal, focused guidance: act
   282‚Üí///    deliberately, verify with `memory_search`, persist lessons with
   283‚Üí///    `memory_add`, decompose with `plan_create` before complex actions.
   284‚Üí///
   285‚Üí/// `context` is interpolated as pretty-printed JSON. If `context` is not
   286‚Üí/// an object, it is wrapped in `{ "context": <value> }` so the prompt
   287‚Üí/// always renders a stable shape.
   288‚Üípub fn build_agent_system_prompt(tool_catalog: &[crate::tools::ToolSpec], context: &Value) -> String {
   289‚Üí    // Render the tool catalog as one block per tool.
   290‚Üí    let mut tool_blocks = String::with_capacity(8 * 1024);
   291‚Üí    for spec in tool_catalog {
   292‚Üí        tool_blocks.push_str(&format!(
   293‚Üí            "### {name}\n{desc}\n\nParameters (JSON Schema):\n```json\n{schema}\n```\n\n",
   294‚Üí            name = spec.name,
   295‚Üí            desc = spec.description,
   296‚Üí            schema = serde_json::to_string_pretty(&spec.parameters)
   297‚Üí                .unwrap_or_else(|_| "{}".to_string()),
   298‚Üí        ));
   299‚Üí    }
   300‚Üí
   301‚Üí    // Render the OS context as pretty JSON. Wrap non-objects so the prompt
   302‚Üí    // shape is stable regardless of caller input.
   303‚Üí    let context_json = if context.is_object() && !context.is_array() {
   304‚Üí        serde_json::to_string_pretty(context).unwrap_or_else(|_| "{}".to_string())
   305‚Üí    } else {
   306‚Üí        serde_json::to_string_pretty(&json!({ "context": context }))
   307‚Üí            .unwrap_or_else(|_| "{}".to_string())
   308‚Üí    };
   309‚Üí
   310‚Üí    format!(
   311‚Üí        "# AETHER AUTONOMOUS AGENT ‚Äî Alpha-OS Cognitive Core\n\
   312‚Üí         You are the autonomous agent core of Alpha-OS, a self-evolving Linux-like OS. \
   313‚Üí         You perceive the current OS state, reason about what to do, and act through tools. \
   314‚Üí         Your job is to achieve the user's goal by issuing tool calls and observing their results, \
   315‚Üí         iterating until the goal is complete.\n\n\
   316‚Üí         ## Goal protocol\n\
   317‚Üí         To call a tool, emit a JSON object anywhere in your response. The object MUST have a \
   318‚Üí         `tool` (or `name`) field set to one of the available tool names, and a `params` (or \
   319‚Üí         `arguments`) field containing the tool's parameters as a JSON object. You may emit \
   320‚Üí         multiple tool calls in a single response; they will be executed in order and their \
   321‚Üí         results fed back to you as `[tool_result: <tool>]` messages.\n\n\
   322‚Üí         Example:\n\
   323‚Üí         ```json\n\
   324‚Üí         {{\n\
   325‚Üí           \"tool\": \"file_read\",\n\
   326‚Üí           \"params\": {{ \"path\": \"/etc/hostname\" }}\n\
   327‚Üí         }}\n\
   328‚Üí         ```\n\n\
   329‚Üí         When ‚Äî and only when ‚Äî you have achieved the user's goal, emit the literal sentinel \
   330‚Üí         `{sentinel}` on its own line. Do not emit it unless the goal is fully achieved.\n\n\
   331‚Üí         ## Available tools\n\
   332‚Üí         {tool_blocks}\n\
   333‚Üí         ## Current OS state\n\
   334‚Üí         The following JSON snapshot describes the current state of the OS you are managing \
   335‚Üí         (open windows, memory, active plan, goals, etc.). Use it to ground your actions.\n\
   336‚Üí         ```json\n\
   337‚Üí         {context_json}\n\
   338‚Üí         ```\n\n\
   339‚Üí         ## Operating principles\n\
   340‚Üí         - Act deliberately: one logical action per iteration when possible.\n\
   341‚Üí         - Verify before trusting: use `memory_search` to recall relevant facts before \
   342‚Üí         destructive actions (`file_delete`, `exec` with side effects).\n\
   343‚Üí         - Persist lessons: use `memory_add` to record anything worth remembering for \
   344‚Üí         future iterations.\n\
   345‚Üí         - Decompose complexity: use `plan_create` for any goal with more than two steps, \
   346‚Üí         then `plan_update` as you complete each step.\n\
   347‚Üí         - Observe carefully: tool results arrive as `[tool_result: <tool>]` messages; \
   348‚Üí         read them before deciding the next action.\n\
   349‚Üí         - Stop only when done: emit `{sentinel}` when the goal is fully achieved.",
   350‚Üí        sentinel = TASK_COMPLETE_SENTINEL,
   351‚Üí        tool_blocks = tool_blocks,
   352‚Üí        context_json = context_json,
   353‚Üí    )
   354‚Üí}

   359‚Üí
   360‚Üí/// Parse zero or more [`ToolCall`]s from the LLM's response text.
   361‚Üí///
   362‚Üí/// Supports every common emission format small GGUF models produce:
   363‚Üí///
   364‚Üí/// - **Inline JSON** ‚Äî `{"tool": "file_read", "params": {"path": "‚Ä¶"}}`
   365‚Üí///   appearing anywhere in the response (including inside prose).
   366‚Üí/// - **Fenced code blocks** ‚Äî ` ```json\n{‚Ä¶}\n``` ` or ` ```tool\n{‚Ä¶}\n``` `.
   367‚Üí///   The fenced content is scanned with the same JSON extractor.
   368‚Üí/// - **OpenAI function-call shape** ‚Äî `{"function": {"name": "‚Ä¶",
   369‚Üí///   "arguments": "‚Ä¶"}}` (arguments may be a JSON string or object).
   370‚Üí/// - **Wrapper arrays** ‚Äî `{"tool_calls": [{‚Ä¶}, {‚Ä¶}]}` expands to
   371‚Üí///   multiple calls.
   372‚Üí/// - **Alternate field names** ‚Äî `name`/`tool_name` for the tool name;
   373‚Üí///   `arguments`/`args` for parameters.
   374‚Üí///
   375‚Üí/// The parser walks the response once, scanning for `{` and attempting to
   376‚Üí/// extract a balanced JSON object at each occurrence. String-literal
   377‚Üí/// braces (e.g. inside a `"```json\n‚Ä¶\n```"` code fence description) are
   378‚Üí/// correctly skipped via the in-string / escape flags. Every successfully
   379‚Üí/// parsed JSON object is then classified by [`value_to_tool_calls`] as a
   380‚Üí/// single call, an array of calls, or a non-tool object (ignored).
   381‚Üí///
   382‚Üí/// Returns the calls in source order. Duplicates are preserved (the LLM
   383‚Üí/// may legitimately emit the same call twice if the first attempt didn't
   384‚Üí/// return the expected result).
   385‚Üípub fn parse_tool_calls(response: &str) -> Vec<ToolCall> {
   386‚Üí    let mut calls = Vec::new();
   387‚Üí    let bytes = response.as_bytes();
   388‚Üí    let mut i = 0;
   389‚Üí    while i < bytes.len() {
   390‚Üí        if bytes[i] == b'{' {
   391‚Üí            if let Some((json_str, end)) = extract_balanced_json(response, i) {
   392‚Üí                if let Ok(v) = serde_json::from_str::<Value>(&json_str) {
   393‚Üí                    calls.extend(value_to_tool_calls(&v));
   394‚Üí                }
   395‚Üí                // Skip past the balanced object whether or not it parsed.
   396‚Üí                // (A failed parse means malformed JSON; we still want to
   397‚Üí                // jump past it to avoid re-scanning its interior.)
   398‚Üí                i = end;
   399‚Üí                continue;
   400‚Üí            }
   401‚Üí        }
   402‚Üí        i += 1;
   403‚Üí    }
   404‚Üí    calls
   405‚Üí}
   406‚Üí
   407‚Üí/// Extract a balanced JSON object (including any nested braces) starting
   408‚Üí/// at `start` (which MUST point at a `{` byte). Returns the matched
   409‚Üí/// substring and the index *after* the closing `}`, or `None` if no
   410‚Üí/// balanced match is found before end-of-input.
   411‚Üí///
   412‚Üí/// Honors JSON string literals (`"‚Ä¶"`) and escape sequences (`\"`, `\\`,
   413‚Üí/// ‚Ä¶) so braces inside strings don't affect depth counting.
   414‚Üífn extract_balanced_json(s: &str, start: usize) -> Option<(String, usize)> {
   415‚Üí    let bytes = s.as_bytes();
   416‚Üí    if start >= bytes.len() || bytes[start] != b'{' {
   417‚Üí        return None;
   418‚Üí    }
   419‚Üí    let mut depth: i32 = 0;
   420‚Üí    let mut in_string = false;
   421‚Üí    let mut escape = false;
   422‚Üí    for i in start..bytes.len() {
   423‚Üí        let c = bytes[i];
   424‚Üí        if in_string {
   425‚Üí            if escape {
   426‚Üí                escape = false;
   427‚Üí            } else if c == b'\\' {
   428‚Üí                escape = true;
   429‚Üí            } else if c == b'"' {
   430‚Üí                in_string = false;
   431‚Üí            }
   432‚Üí        } else {
   433‚Üí            match c {
   434‚Üí                b'"' => in_string = true,
   435‚Üí                b'{' => depth += 1,
   436‚Üí                b'}' => {
   437‚Üí                    depth -= 1;
   438‚Üí                    if depth == 0 {
   439‚Üí                        return Some((s[start..=i].to_string(), i + 1));
   440‚Üí                    }
   441‚Üí                }
   442‚Üí                _ => {}
   443‚Üí            }
   444‚Üí        }
   445‚Üí    }
   446‚Üí    None
   447‚Üí}
   448‚Üí
   449‚Üí/// Classify a parsed JSON value as zero or more tool calls. Handles single
   450‚Üí/// calls, call arrays, and the `{"tool_calls": [...]}` wrapper.
   451‚Üífn value_to_tool_calls(v: &Value) -> Vec<ToolCall> {
   452‚Üí    // Array of calls: [{...}, {...}]
   453‚Üí    if let Some(arr) = v.as_array() {
   454‚Üí        return arr.iter().filter_map(value_to_single_call).collect();
   455‚Üí    }
   456‚Üí    // Wrapper: {"tool_calls": [...]} or {"calls": [...]}
   457‚Üí    for wrapper_key in ["tool_calls", "calls"] {
   458‚Üí        if let Some(inner) = v.get(wrapper_key).and_then(|x| x.as_array()) {
   459‚Üí            return inner.iter().filter_map(value_to_single_call).collect();
   460‚Üí        }
   461‚Üí    }
   462‚Üí    // Single call
   463‚Üí    if let Some(c) = value_to_single_call(v) {
   464‚Üí        return vec![c];
   465‚Üí    }
   466‚Üí    Vec::new()
   467‚Üí}
   468‚Üí
   469‚Üí/// Convert a single JSON object to a [`ToolCall`]. Accepts the field-name
   470‚Üí/// aliases documented on [`parse_tool_calls`]. Returns `None` when the
   471‚Üí/// object doesn't look like a tool call (no `tool`/`name`/`tool_name`/
   472‚Üí/// `function.name` field).
   473‚Üífn value_to_single_call(v: &Value) -> Option<ToolCall> {
   474‚Üí    // Direct format: {"tool": "...", "params": {...}}
   475‚Üí    let name = v
   476‚Üí        .get("tool")
   477‚Üí        .and_then(|x| x.as_str())
   478‚Üí        .or_else(|| v.get("name").and_then(|x| x.as_str()))
   479‚Üí        .or_else(|| v.get("tool_name").and_then(|x| x.as_str()));
   480‚Üí
   481‚Üí    if let Some(name) = name {
   482‚Üí        let params = v
   483‚Üí            .get("params")
   484‚Üí            .cloned()
   485‚Üí            .or_else(|| v.get("arguments").cloned())
   486‚Üí            .or_else(|| v.get("args").cloned())
   487‚Üí            .filter(Value::is_object)
   488‚Üí            .unwrap_or_else(|| json!({}));
   489‚Üí        return Some(ToolCall {
   490‚Üí            name: name.to_string(),
   491‚Üí            params,
   492‚Üí        });
   493‚Üí    }
   494‚Üí
   495‚Üí    // OpenAI function-call format: {"id": "...", "function": {"name": "...", "arguments": "..."}}
   496‚Üí    if let Some(func) = v.get("function") {
   497‚Üí        if let Some(fname) = func.get("name").and_then(|x| x.as_str()) {
   498‚Üí            let params = match func.get("arguments") {
   499‚Üí                // OpenAI ships arguments as a JSON *string* ‚Äî parse it.
   500‚Üí                Some(Value::String(s)) => {
   501‚Üí                    serde_json::from_str(s).unwrap_or_else(|_| json!({}))
   502‚Üí                }
   503‚Üí                Some(other) if other.is_object() => other.clone(),
   504‚Üí                _ => json!({}),
   505‚Üí            };
   506‚Üí            return Some(ToolCall {
   507‚Üí                name: fname.to_string(),
   508‚Üí                params,
   509‚Üí            });
   510‚Üí        }
   511‚Üí    }
   512‚Üí
   513‚Üí    None
   514‚Üí}

   520‚Üí/// Send the current message stream to the backend LLM and return the
   521‚Üí/// assistant's response text.
   522‚Üí///
   523‚Üí/// Uses the same `state.backend` and `state.client` as the rest of the
   524‚Üí/// engine, so all backend configuration (timeouts, retries, base URL)
   525‚Üí/// applies. On any failure (network error, non-2xx, empty body, malformed
   526‚Üí/// JSON), returns a `[aether-agent] ‚Ä¶` diagnostic string so the agent
   527‚Üí/// loop can surface the problem and stop gracefully.
   528‚Üíasync fn call_llm(state: &AppState, messages: &[AgentMessage]) -> String {
   529‚Üí    let request_body = json!({
   530‚Üí        "messages": messages.iter().map(AgentMessage::to_request_json).collect::<Vec<_>>(),
   531‚Üí        "temperature": 0.7,
   532‚Üí    });
   533‚Üí    let url = format!(
   534‚Üí        "{}/chat/completions",
   535‚Üí        state.backend.trim_end_matches('/')
   536‚Üí    );
   537‚Üí
   538‚Üí    match state.client.post(&url).json(&request_body).send().await {
   539‚Üí        Ok(resp) if resp.status().is_success() => {
   540‚Üí            let txt = resp.text().await.unwrap_or_default();
   541‚Üí            extract_assistant_content(&txt).unwrap_or_else(|| {
   542‚Üí                format!(
   543‚Üí                    "[aether-agent] empty backend response: {}",
   544‚Üí                    txt.chars().take(200).collect::<String>()
   545‚Üí                )
   546‚Üí            })
   547‚Üí        }
   548‚Üí        Ok(resp) => {
   549‚Üí            let status = resp.status();
   550‚Üí            let body_txt = resp.text().await.unwrap_or_default();
   551‚Üí            format!(
   552‚Üí                "[aether-agent] backend returned {status}: {}",
   553‚Üí                body_txt.chars().take(200).collect::<String>()
   554‚Üí            )
   555‚Üí        }
   556‚Üí        Err(e) => format!("[aether-agent] backend unreachable: {e}"),
   557‚Üí    }
   558‚Üí}
   559‚Üí
   560‚Üí/// Extract the assistant's text content from an OpenAI-shaped chat
   561‚Üí/// completion response body. Mirrors the same helper in `handlers.rs` but
   562‚Üí/// kept local to this module to avoid widening the visibility of the
   563‚Üí/// existing private function (which would break the "no signature changes"
   564‚Üí/// constraint).
   565‚Üífn extract_assistant_content(body: &str) -> Option<String> {
   566‚Üí    let v: Value = serde_json::from_str(body).ok()?;
   567‚Üí    v.get("choices")
   568‚Üí        .and_then(|c| c.get(0))
   569‚Üí        .and_then(|c| c.get("message"))
   570‚Üí        .and_then(|m| m.get("content"))
   571‚Üí        .and_then(|c| c.as_str())
   572‚Üí        .map(|s| s.to_string())
   573‚Üí}

   576‚Üí// The main agent loop
   577‚Üí// ---------------------------------------------------------------------------
   578‚Üí
   579‚Üí/// Run the autonomous agent loop until termination.
   580‚Üí///
   581‚Üí/// # Arguments
   582‚Üí///
   583‚Üí/// - `state` ‚Äî the engine's shared [`AppState`] (used for the backend LLM
   584‚Üí///   call only; no other engine state is touched).
   585‚Üí/// - `goal` ‚Äî the user's natural-language goal (e.g. *"set up a Python
   586‚Üí///   dev environment and write a hello-world script"*).
   587‚Üí/// - `context` ‚Äî the current OS state as a JSON object (open windows,
   588‚Üí///   memory snapshot, active plan, goals, ‚Ä¶). Pretty-printed into the
   589‚Üí///   system prompt so the LLM perceives it.
   590‚Üí/// - `max_iterations` ‚Äî iteration cap. Clamped to `[1, ABSOLUTE_MAX_ITERATIONS]`.
   591‚Üí///
   592‚Üí/// # Returns
   593‚Üí///
   594‚Üí/// An [`AgentRunResult`] carrying the final assistant text, the iteration
   595‚Üí/// count, the full transcript of tool calls, and the `completed` flag.
   596‚Üí///
   597‚Üí/// # Termination
   598‚Üí///
   599‚Üí/// See the module docs. The loop stops on `TASK_COMPLETE`, on a
   600‚Üí/// no-tool-calls response, or on hitting `max_iterations`.
   601‚Üí///
   602‚Üí/// # Errors
   603‚Üí///
   604‚Üí/// This function never returns `Result::Err` ‚Äî every failure path
   605‚Üí/// (backend unreachable, malformed response, unknown tool, invalid
   606‚Üí/// params, executor error) is converted into a diagnostic string and fed
   607‚Üí/// back into the message stream or surfaced in the final result. The
   608‚Üí/// caller decides how to react.
   609‚Üípub async fn run_agent_loop(
   610‚Üí    state: AppState,
   611‚Üí    goal: &str,
   612‚Üí    context: &Value,
   613‚Üí    max_iterations: usize,
   614‚Üí) -> AgentRunResult {
   615‚Üí    // Clamp the iteration cap. We accept the caller's value but enforce a
   616‚Üí    // hard ceiling so a buggy caller can't pin a CPU forever against a
   617‚Üí    // broken backend.
   618‚Üí    let cap = max_iterations.clamp(1, ABSOLUTE_MAX_ITERATIONS);
   619‚Üí
   620‚Üí    // Build the tool registry + system prompt once; both are reused on
   621‚Üí    // every iteration.
   622‚Üí    let registry = ToolRegistry::new();
   623‚Üí    let catalog = registry.catalog();
   624‚Üí    let system_prompt = build_agent_system_prompt(&catalog, context);
   625‚Üí
   626‚Üí    // Initial state: [system, user(goal)].
   627‚Üí    let mut agent_state = AgentState {
   628‚Üí        iteration: 0,
   629‚Üí        messages: vec![
   630‚Üí            AgentMessage::system(system_prompt),
   631‚Üí            AgentMessage::user(goal.to_string()),
   632‚Üí        ],
   633‚Üí        tool_results: Vec::new(),
   634‚Üí        completed: false,
   635‚Üí    };
   636‚Üí
   637‚Üí    let mut all_tool_calls: Vec<ToolCall> = Vec::new();
   638‚Üí    let mut final_text = String::new();
   639‚Üí
   640‚Üí    for i in 0..cap {
   641‚Üí        agent_state.iteration = i + 1;
   642‚Üí
   643‚Üí        // ---- 2. Think: call the LLM with the current message stream ----
   644‚Üí        let response_text = call_llm(&state, &agent_state.messages).await;
   645‚Üí
   646‚Üí        // ---- 3a. Check for explicit completion sentinel ----
   647‚Üí        if response_text.contains(TASK_COMPLETE_SENTINEL) {
   648‚Üí            final_text = response_text;
   649‚Üí            agent_state.completed = true;
   650‚Üí            break;
   651‚Üí        }
   652‚Üí
   653‚Üí        // ---- 3b. Parse tool calls from the response ----
   654‚Üí        let calls = parse_tool_calls(&response_text);
   655‚Üí        all_tool_calls.extend(calls.clone());
   656‚Üí
   657‚Üí        // ---- 3c. No tool calls + no sentinel ‚Üí stop (incomplete) ----
   658‚Üí        if calls.is_empty() {
   659‚Üí            final_text = response_text;
   660‚Üí            // `completed` stays false ‚Äî the agent stopped without signaling.
   661‚Üí            break;
   662‚Üí        }
   663‚Üí
   664‚Üí        // Record the assistant message that contained the tool calls, so
   665‚Üí        // the next think step sees the full conversation context.
   666‚Üí        agent_state.messages.push(AgentMessage::assistant(response_text.clone()));
   667‚Üí
   668‚Üí        // ---- 4. Execute: dispatch each tool call through the registry ----
   669‚Üí        for call in &calls {
   670‚Üí            let (output, success) = match Tool::from_request(&call.name, &call.params) {
   671‚Üí                Some(tool) => registry.execute(&tool).await,
   672‚Üí                None => (
   673‚Üí                    format!(
   674‚Üí                        "tool_not_found: no tool named \"{}\" in registry. \
   675‚Üí                         Available tools: {}",
   676‚Üí                        call.name,
   677‚Üí                        catalog
   678‚Üí                            .iter()
   679‚Üí                            .map(|t| t.name)
   680‚Üí                            .collect::<Vec<_>>()
   681‚Üí                            .join(", ")
   682‚Üí                    ),
   683‚Üí                    false,
   684‚Üí                ),
   685‚Üí            };
   686‚Üí
   687‚Üí            let result = ToolResult {
   688‚Üí                tool: call.name.clone(),
   689‚Üí                output: output.clone(),
   690‚Üí                success,
   691‚Üí            };
   692‚Üí            agent_state.tool_results.push(result);
   693‚Üí
   694‚Üí            // ---- 5. Observe: feed the tool result back into the stream ----
   695‚Üí            agent_state
   696‚Üí                .messages
   697‚Üí                .push(AgentMessage::tool(call.name.clone(), output));
   698‚Üí        }
   699‚Üí
   700‚Üí        // The "final text" is the most recent assistant response, so if
   701‚Üí        // the next iteration hits the iteration cap we still return
   702‚Üí        // something useful.
   703‚Üí        final_text = response_text;
   704‚Üí
   705‚Üí        // ---- 6. Repeat (loop continues to the next think step) ----
   706‚Üí    }
   707‚Üí
   708‚Üí    AgentRunResult {
   709‚Üí        result: final_text,
   710‚Üí        iterations: agent_state.iteration,
   711‚Üí        tool_calls: all_tool_calls,
   712‚Üí        completed: agent_state.completed,
   713‚Üí    }
   714‚Üí}
