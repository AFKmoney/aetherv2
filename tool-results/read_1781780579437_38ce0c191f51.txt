     1→//! # HTTP Handlers — the 10-stage Cognitive Pipeline (Aether Engine v3.0)
     2→//!
     3→//! All Axum route handlers live here. The flagship handler,
     4→//! [`chat_completions`], implements the full cognitive pipeline that gives
     5→//! Aether Engine its 10x capacity multiplier over a vanilla GGUF backend:
     6→//!
     7→//!   1. **Action cache check** — instant for repeated/similar queries
     8→//!      ([`crate::cache::ActionCache`]).
     9→//!   2. **Graph retrieval** — TF-IDF semantic search + 1-hop edge expansion
    10→//!      ([`crate::graph::MemoryGraph::retrieve`]).
    11→//!   3. **Context compression** — 40K→4K tokens preserving signal
    12→//!      ([`crate::compress::compress`]).
    13→//!   4. **Complexity analysis** — Simple / Moderate / Complex
    14→//!      ([`crate::decompose::analyze_complexity`]).
    15→//!   5. **Cognitive decomposition** — Complex queries are broken into
    16→//!      sub-questions ([`crate::decompose::decompose`]).
    17→//!   6. **Sequential sub-question solving** — each with fresh context +
    18→//!      dependency injection (driven by this module via [`call_backend`]).
    19→//!   7. **Synthesis** — combine sub-answers into final response (the
    20→//!      `"synth"` sub-question, or a plain concatenation fallback).
    21→//!   8. **ATD verification** — dual-graph likelihood-vs-entropy collision;
    22→//!      retry once if failed ([`crate::atd::verify`]).
    23→//!   9. **Knowledge distillation** — store successful decomposition patterns
    24→//!      ([`crate::decompose::DistillationStore`]).
    25→//!  10. **Speculative prefetch** — warm the retrieval cache for
    26→//!      graph-adjacent queries so the next related query is instant
    27→//!      ([`prefetch`]).
    28→//!
    29→//! # OpenAI compatibility
    30→//!
    31→//! [`chat_completions`] mirrors the OpenAI `POST /v1/chat/completions`
    32→//! contract (request body in, chat-completion-shaped JSON out) so the
    33→//! engine can be dropped in as a drop-in replacement for any OpenAI-compatible
    34→//! client. The backend it forwards to is *also* OpenAI-compatible (default
    35→//! `http://localhost:11434/v1`, i.e. Ollama).

    71→///
    72→/// Returns the engine version, graph/cache/distillation sizes, and the full

    85→        let h = state.hcm.lock().await;
    86→        (h.pair_count, h.interference(), h.memory_bytes(), h.capacity())
    87→    };

    89→        "ok": true,
    90→        "version": "3.0",
    91→        "nodes": nodes,
    92→        "edges": edges,
    93→        "cache_hits": stats.cache_hits,
    94→        "cache_entries": cache_count,
    95→        "decompositions": stats.decompositions,
    96→        "verifications": stats.verifications,
    97→        "verifications_passed": stats.verifications_passed,
    98→        "distillation_patterns": distillation_count,
    99→        "distillation_hits": stats.distillation_hits,
   100→        "requests": stats.requests,
   101→        "hcm_pairs_folded": stats.hcm_pairs_folded,
   102→        "hcm_probes": stats.hcm_probes,
   103→        "hcm_active_pairs": hcm_pairs,
   104→        "hcm_interference": (hcm_interference * 100.0).round() / 100.0,
   105→        "hcm_memory_bytes": hcm_memory,
   106→        "hcm_capacity": hcm_capacity,

   111→        "atd_validated": stats.atd_validated,
   112→        "atd_rejected": stats.atd_rejected,

   163→// ---------------------------------------------------------------------------
   164→// POST /graph/add
   165→// ---------------------------------------------------------------------------
   166→
   167→/// Add (or replace) a node in the semantic memory graph.
   168→///
   169→/// Side-effect: the new node is *also* folded into the Holographic Context
   170→/// Memory arena as `(hash(id), hash(text))` so the HCM has a parallel
   171→/// associative copy of every graph memory. The fold counter in
   172→/// [`Stats::hcm_pairs_folded`](crate::Stats) is bumped accordingly.
   173→///
   174→/// Returns `{ ok, id, edges_created }` where `edges_created` is the total
   175→/// number of adjacency entries created across the whole graph by the
   176→/// recompute (see [`MemoryGraph::add`](crate::graph::MemoryGraph::add)).
   177→pub async fn graph_add(
   178→    State(state): State<AppState>,
   179→    Json(req): Json<AddNodeRequest>,
   180→) -> impl IntoResponse {
   181→    let id = req.id.clone();
   182→    let text = req.text.clone();
   183→    let edges_created = {
   184→        let mut g = state.graph.lock().await;
   185→        g.add(req)
   186→    };
   187→    // Also fold into the Holographic Context Memory (HCM)
   188→    {
   189→        let mut hcm = state.hcm.lock().await;
   190→        let key = crate::hcm::hash_to_vector(&id, hcm.dim);
   191→        let value = crate::hcm::hash_to_vector(&text, hcm.dim);
   192→        hcm.fold(&key, &value);
   193→    }
   194→    state.stats.lock().await.hcm_pairs_folded += 1;
   195→    Json(json!({ "ok": true, "id": id, "edges_created": edges_created }))
   196→}

   217→/// Pure semantic search (no edge expansion). Returns the top-`limit`
   218→/// nodes by cosine similarity to the query. Differs from the pipeline's
   219→/// Stage-2 retrieval in that it does *not* do the 1-hop edge expansion —
   220→/// callers that want raw TF-IDF hits use this; the pipeline uses
   221→/// [`MemoryGraph::retrieve`](crate::graph::MemoryGraph::retrieve).
   222→pub async fn graph_search(
   223→    State(state): State<AppState>,
   224→    Json(req): Json<SearchRequest>,
   225→) -> impl IntoResponse {
   226→    let results: Vec<ScoredNode> = {
   227→        let g = state.graph.lock().await;
   228→        g.search(&req.query, req.limit)
   229→    };
   230→    Json(json!({ "query": req.query, "results": results }))
   231→}

   251→// ---------------------------------------------------------------------------
   252→// POST /v1/interrupt — interrupt the current inference and inject new input
   253→//
   254→// Allows the user to interrupt the AI mid-reasoning and redirect it.
   255→// The request body contains:
   256→//   { "action": "interrupt", "new_instruction": "user's new directive" }
   257→//
   258→// When interrupted, the current pipeline's backend call is aborted (if in
   259→// progress) and the new instruction is queued for the next cycle.
   260→// ---------------------------------------------------------------------------
   261→
   262→/// Interrupt the current inference and inject a new directive.
   263→///
   264→/// Accepts a JSON body of the shape `{ "action": "interrupt", "new_instruction": "…" }`.
   265→/// Two `action` values are supported:
   266→///
   267→/// - `"interrupt"` — injects `new_instruction` into the memory graph as
   268→///   a high-priority `"interrupt"` node so the next retrieval picks it up.
   269→///   A full implementation would also cancel the in-flight backend request
   270→///   via an `AbortHandle`; the current implementation queues the directive
   271→///   for the next cycle (see the inline comment in the handler body).
   272→/// - `"status"` — returns whether an inference is currently in progress
   273→///   (always `false` in this version; would be `true` if a pipeline is
   274→///   running once the abort-handle wiring is added).
   275→///
   276→/// Any other `action` value yields `{ ok: false, error: "unknown action" }`.
   277→pub async fn interrupt(
   278→    State(state): State<AppState>,
   279→    Json(body): Json<serde_json::Value>,
   280→) -> impl IntoResponse {
   281→    let action = body.get("action").and_then(|a| a.as_str()).unwrap_or("interrupt");
   282→    let new_instruction = body.get("new_instruction").and_then(|i| i.as_str()).unwrap_or("");
   283→
   284→    match action {
   285→        "interrupt" => {
   286→            // In a full implementation, this would cancel the in-flight backend
   287→            // request via an AbortHandle. For now, we set a flag that the next
   288→            // pipeline check will see, and inject the new instruction as a
   289→            // graph node so it's available for the next retrieval.
   290→            if !new_instruction.is_empty() {
   291→                let mut g = state.graph.lock().await;
   292→                let id = format!("interrupt-{}", std::time::SystemTime::now()
   293→                    .duration_since(std::time::UNIX_EPOCH)
   294→                    .map(|d| d.as_millis())
   295→                    .unwrap_or(0));
   296→                g.add(crate::graph::AddNodeRequest {
   297→                    id,
   298→                    text: new_instruction.to_string(),
   299→                    kind: "interrupt".to_string(),
   300→                    metadata: serde_json::json!({ "source": "user", "priority": "high" }),
   301→                });
   302→            }
   303→
   304→            Json(json!({
   305→                "ok": true,
   306→                "interrupted": true,
   307→                "message": "Inference interrupted. New instruction injected into the memory graph for the next cycle.",
   308→                "new_instruction": new_instruction,
   309→            }))
   310→        }
   311→        "status" => {
   312→            // Check if an inference is currently in progress
   313→            let stats = state.stats.lock().await;
   314→            Json(json!({
   315→                "ok": true,
   316→                "requests": stats.requests,
   317→                "active": false, // Would be true if a pipeline is running
   318→            }))
   319→        }
   320→        _ => {
   321→            Json(json!({ "ok": false, "error": "unknown action" }))
   322→        }
   323→    }
   324→}

   326→// ---------------------------------------------------------------------------
   327→// POST /v1/chat/completions — the full cognitive pipeline (OpenAI-compatible)
   328→// ---------------------------------------------------------------------------
   329→
   330→/// OpenAI-compatible chat-completions endpoint that runs the full 10-stage
   331→/// cognitive pipeline.
   332→///
   333→/// This is the heart of Aether Engine. It accepts an OpenAI-shaped
   334→/// `messages` array, augments it through the pipeline (cache → graph →
   335→/// compress → complexity → decompose → solve → synthesize → ATD-verify →
   336→/// distill → cache → prefetch), and returns an OpenAI-shaped
   337→/// `chat.completion` JSON response.
   338→///
   339→/// # Complexity branching
   340→///
   341→/// - **Simple** → one backend call.
   342→/// - **Moderate** → two backend calls: a think step, then an answer step
   343→///   that consumes the think step's output.
   344→/// - **Complex** → check the distillation store for a reusable
   345→///   decomposition pattern; if hit, reuse it; otherwise run [`decompose`].
   346→///   Solve each sub-question sequentially with dependency injection, then
   347→///   synthesize.
   348→///
   349→/// # ATD retry
   350→///
   351→/// Stage 8 runs [`atd::verify`] on the synthesized response. If it fails,
   352→/// a single retry is attempted with a recommendation-specific prompt
   353→/// injection (`FallBackToSimpleShot` ⇒ clean simple retry; anything else
   354→/// ⇒ standard retry prompt). If the retry also fails ATD, the response
   355→/// with the better `collision_delta` is returned.
   356→///
   357→/// # Caching & prefetch
   358→///
   359→/// Stage 1 checks the action cache (instant return on hit). The final
   360→/// response is always written back to the action cache. Stage 10 spawns a
   361→/// detached [`prefetch`] task that warms the retrieval cache for the
   362→/// top-3 retrieved nodes' texts so a follow-up question on the same
   363→/// subgraph is fast.
   364→pub async fn chat_completions(
   365→    State(state): State<AppState>,
   366→    Json(body): Json<serde_json::Value>,
   367→) -> impl IntoResponse {
   368→    let start = Instant::now();
   369→    state.stats.lock().await.requests += 1;
   370→
   371→    // Extract the last user message as the retrieval query.
   372→    let query = extract_user_query(&body);
   373→
   374→    // Vectorize the query against the graph's current vocabulary.
   375→    let qvec: SparseVec = {
   376→        let g = state.graph.lock().await;
   377→        g.vectorizer.vectorize(&query)
   378→    };
   379→
   380→    // ---- Stage 1: Action cache fast-path (similarity > 0.95) ----
   381→    // The cache is checked *before* any backend work — a hit short-circuits
   382→    // the entire pipeline and returns instantly.
   383→    {
   384→        let cache = state.cache.lock().await;
   385→        if let Some((resp, _sim)) = cache.get(&query, &qvec) {
   386→            // Explicitly drop the cache guard *before* re-locking stats to
   387→            // avoid holding two mutexes simultaneously (which would be a
   388→            // deadlock risk if the lock order were ever reversed elsewhere).
   389→            drop(cache);
   390→            state.stats.lock().await.cache_hits += 1;
   391→            return Json(openai_completion(&resp, "aether-cache")).into_response();
   392→        }
   393→    }
   394→
   395→    // ---- Stage 2: Graph retrieval (TF-IDF + edge expansion) ----
   396→    let retrieved: Vec<ScoredNode> = {
   397→        let g = state.graph.lock().await;
   398→        g.retrieve(&query, 8)
   399→    };
   400→
   401→    // ---- Stage 3: Context compression (40K→4K preserving signal) ----
   402→    // Two-tier: hit the retrieval cache first (warmed by the prefetcher);
   403→    // on miss, run the compressor and write the result back so the next
   404→    // similar query skips this expensive step.
   405→    let context_block = {
   406→        let mut cache = state.cache.lock().await;
   407→        if let Some(ctx) = cache.get_retrieval(&query, &qvec) {
   408→            ctx
   409→        } else {
   410→            let ctx = compress::compress(&retrieved, &query, 6000);
   411→            cache.put_retrieval(&query, qvec.clone(), ctx.clone());
   412→            ctx
   413→        }
   414→    };
   415→
   416→    // ---- Stage 4: Complexity analysis ----
   417→    let complexity = decompose::analyze_complexity(&query);
   418→    let mut pipeline = PipelineState::new(&query);
   419→    pipeline.complexity = complexity.clone();
   420→
   421→    // ---- Stage 5-7: Cognitive decomposition + sequential solving + synthesis ----
   422→    let final_response = match complexity {
   423→        Complexity::Simple => {
   424→            // Simple query — one-shot through the backend
   425→            pipeline.stages_completed.push("one-shot".into());
   426→            call_backend(&state, &body, &context_block).await
   427→        }
   428→        Complexity::Moderate => {
   429→            // Moderate — two-step: think, then answer
   430→            state.stats.lock().await.decompositions += 1;
   431→            pipeline.stages_completed.push("two-step-think".into());
   432→            pipeline.stages_completed.push("two-step-answer".into());
   433→
   434→            // Step 1: Think about the problem
   435→            let think_body = inject_think_step(&body, &context_block, &query);
   436→            let think_response = call_backend(&state, &think_body, &context_block).await;
   437→            pipeline.total_backend_calls += 1;
   438→
   439→            // Step 2: Answer using the thinking result
   440→            let answer_body = inject_answer_step(&body, &think_response, &context_block);
   441→            call_backend(&state, &answer_body, &context_block).await
   442→        }
   443→        Complexity::Complex => {
   444→            // Complex — full decomposition pipeline
   445→            state.stats.lock().await.decompositions += 1;
   446→            pipeline.stages_completed.push("decompose".into());
   447→
   448→            // Check distillation store for a reusable decomposition pattern
   449→            let sub_questions = {
   450→                let distill = state.distillation.lock().await;
   451→                if let Some(subs) = distill.find(&query, &qvec) {
   452→                    state.stats.lock().await.distillation_hits += 1;
   453→                    pipeline.stages_completed.push("distillation-hit".into());
   454→                    subs
   455→                } else {
   456→                    // Fresh decomposition
   457→                    decompose::decompose(&query, &retrieved)
   458→                }
   459→            };
   460→
   461→            pipeline.sub_questions = sub_questions.clone();
   462→            pipeline.stages_completed.push("solve".into());
   463→
   464→            // Solve each sub-question sequentially, injecting dependency
   465→            // answers. The decomposer emits sub-questions in dependency
   466→            // order (topological), so by the time we reach sub-question N
   467→            // every entry it depends_on is already in the `answers` map.
   468→            // The backend sees only one sub-question at a time, with a
   469→            // fresh window — that's the whole point of decomposition:
   470→            // the model never has to hold more than one step in its head.
   471→            let mut answers: HashMap<String, String> = HashMap::new();
   472→            for sub in &sub_questions {
   473→                let sub_prompt = decompose::build_sub_prompt(sub, &answers);
   474→                let sub_body = inject_sub_question(&body, &sub_prompt, &context_block);
   475→                let sub_answer = call_backend(&state, &sub_body, &context_block).await;
   476→                pipeline.total_backend_calls += 1;
   477→                answers.insert(sub.id.clone(), sub_answer);
   478→            }
   479→
   480→            // Synthesis: if there's a "synth" question, its answer is the final response
   481→            pipeline.stages_completed.push("synthesize".into());
   482→            if sub_questions.iter().any(|s| s.id == "synth") {
   483→                answers.get("synth").cloned().unwrap_or_default()
   484→            } else {
   485→                // No explicit synthesis — combine all answers
   486→                let combined: Vec<String> = sub_questions
   487→                    .iter()
   488→                    .filter_map(|s| answers.get(&s.id).cloned())
   489→                    .collect();
   490→                combined.join("\n\n")
   491→            }
   492→        }
   493→    };
   494→
   495→    pipeline.synthesis = Some(final_response.clone());
   496→
   497→    // ---- Stage 8: Asymmetric Tensor Dueling (ATD) verification ----
   498→    // Replaces the simple verify_response with a dual-graph collision:
   499→    // Graph A (likelihood) must overcome Graph B (entropy) for validation.
   500→    state.stats.lock().await.verifications += 1;
   501→    state.stats.lock().await.atd_verifications += 1;
   502→
   503→    let atd_config = crate::atd::ATDConfig::default();
   504→    let atd_result = crate::atd::verify(&final_response, &query, &atd_config);
   505→
   506→    let final_output = if atd_result.validated {
   507→        // ATD validated — response survived the likelihood-entropy collision
   508→        state.stats.lock().await.verifications_passed += 1;
   509→        state.stats.lock().await.atd_validated += 1;
   510→        pipeline.verification_passed = true;
   511→        pipeline.stages_completed.push("atd-validated".into());
   512→        final_response
   513→    } else {
   514→        // ATD rejected — retry based on the recommendation
   515→        state.stats.lock().await.atd_rejected += 1;
   516→        pipeline.stages_completed.push("atd-rejected-retry".into());
   517→
   518→        let retry_body = match atd_result.recommendation {
   519→            crate::atd::ATDRecommendation::FallBackToSimpleShot => {
   520→                // Model is confused — simplify
   521→                inject_simple_retry(&body, &context_block, &query)
   522→            }
   523→            _ => {
   524→                // Standard retry with more explicit prompt
   525→                inject_retry(&body, &context_block, &query)
   526→            }
   527→        };
   528→        let retry_response = call_backend(&state, &retry_body, &context_block).await;
   529→        pipeline.total_backend_calls += 1;
   530→
   531→        // Re-verify the retry with ATD
   532→        let retry_atd = crate::atd::verify(&retry_response, &query, &atd_config);
   533→        if retry_atd.validated {
   534→            state.stats.lock().await.verifications_passed += 1;
   535→            state.stats.lock().await.atd_validated += 1;
   536→            pipeline.verification_passed = true;
   537→            pipeline.stages_completed.push("atd-retry-validated".into());
   538→            retry_response
   539→        } else {
   540→            // Both attempts failed ATD — return the one with better collision delta
   541→            state.stats.lock().await.atd_rejected += 1;
   542→            pipeline.verification_passed = false;
   543→            pipeline.stages_completed.push("atd-fail-final".into());
   544→            if retry_atd.collision_delta > atd_result.collision_delta {
   545→                retry_response
   546→            } else {
   547→                final_response
   548→            }
   549→        }
   550→    };
   551→
   552→    pipeline.total_latency_ms = start.elapsed().as_millis() as u64;
   553→
   554→    // ---- Stage 9: Knowledge distillation (store successful patterns) ----
   555→    // Only Complex queries get a reusable decomposition pattern. Moderate
   556→    // queries use a fixed two-step template that doesn't benefit from
   557→    // caching, and Simple queries don't decompose at all. The pattern is
   558→    // stored only when ATD verification passed — we don't want to
   559→    // distill a *failed* decomposition (it would poison future lookups).
   560→    if matches!(complexity, Complexity::Complex) && pipeline.verification_passed {
   561→        let mut distill = state.distillation.lock().await;
   562→        distill.store(&query, qvec.clone(), pipeline.sub_questions.clone());
   563→    }
   564→
   565→    // ---- Cache the final response ----
   566→    state.cache.lock().await.put(&query, qvec.clone(), final_output.clone());
   567→
   568→    // ---- Stage 10: Speculative prefetch ----
   569→    prefetch(state.clone(), &retrieved).await;
   570→
   571→    Json(openai_completion(&final_output, "aether-pipeline")).into_response()
   572→}

   578→/// Forward a request to the GGUF backend, returning the assistant's text response.
   579→/// Falls back to a graph-only response if the backend is unreachable.
   580→async fn call_backend(
   581→    state: &AppState,
   582→    body: &serde_json::Value,
   583→    context_block: &str,
   584→) -> String {
   585→    let augmented = augment_messages(body, context_block);
   586→    let backend_url = format!("{}/chat/completions", state.backend.trim_end_matches('/'));
   587→    let client = state.client.clone();
   588→
   589→    match client.post(&backend_url).json(&augmented).send().await {
   590→        Ok(resp) if resp.status().is_success() => {
   591→            let txt = resp.text().await.unwrap_or_else(|_| "{}".to_string());
   592→            extract_assistant_content(&txt).unwrap_or_else(|| {
   593→                fallback_response(context_block, "empty backend response")
   594→            })
   595→        }
   596→        Ok(resp) => {
   597→            let status = resp.status();
   598→            let body_txt = resp.text().await.unwrap_or_default();
   599→            fallback_response(
   600→                context_block,
   601→                &format!("backend returned {status}: {}", body_txt.chars().take(200).collect::<String>()),
   602→            )
   603→        }
   604→        Err(e) => {
   605→            fallback_response(context_block, &format!("backend unreachable: {e}"))
   606→        }
   607→    }
   608→}

   614→/// Inject the Aether context block into the system message.
   615→fn augment_messages(body: &serde_json::Value, context_block: &str) -> serde_json::Value {
   616→    let mut out = body.clone();
   617→    let prompt = aether_prompt(context_block);
   618→    if let Some(messages) = out.get_mut("messages").and_then(|m| m.as_array_mut()) {
   619→        let mut injected = false;
   620→        for m in messages.iter_mut() {
   621→            if m.get("role").and_then(|r| r.as_str()) == Some("system") && !injected {
   622→                let existing = m
   623→                    .get("content")
   624→                    .and_then(|c| c.as_str())
   625→                    .unwrap_or("")
   626→                    .to_string();
   627→                let new_content = format!("{}\n\n{}", prompt, existing);
   628→                if let Some(obj) = m.as_object_mut() {
   629→                    obj.insert("content".to_string(), serde_json::Value::String(new_content));
   630→                }
   631→                injected = true;
   632→            }
   633→        }
   634→        if !injected {
   635→            messages.insert(0, json!({ "role": "system", "content": prompt }));
   636→        }
   637→    }
   638→    out
   639→}

   672→/// Inject a sub-question as the primary user message for decomposition.
   673→fn inject_sub_question(body: &serde_json::Value, sub_prompt: &str, context_block: &str) -> serde_json::Value {
   674→    let mut out = body.clone();
   675→    let sub_system = format!(
   676→        "{context_block}\n\n\
   677→         # AETHER COGNITIVE PIPELINE — SUB-QUESTION\n\
   678→         You are solving one step of a larger problem. Focus ONLY on this sub-question.\n\
   679→         Provide a clear, concise answer."
   680→    );
   681→    if let Some(messages) = out.get_mut("messages").and_then(|m| m.as_array_mut()) {

   683→        if let Some(sys) = messages.get_mut(0) {
   684→            if sys.get("role").and_then(|r| r.as_str()) == Some("system") {
   685→                if let Some(obj) = sys.as_object_mut() {
   686→                    obj.insert("content".to_string(), serde_json::Value::String(sub_system));
   687→                }
   688→            }
   689→        }
   690→        // Replace the last user message with the sub-question
   691→        if let Some(last) = messages.last_mut() {
   692→            if last.get("role").and_then(|r| r.as_str()) == Some("user") {
   693→                if let Some(obj) = last.as_object_mut() {
   694→                    obj.insert("content".to_string(), serde_json::Value::String(sub_prompt.to_string()));
   695→                }
   696→            }
   697→        }
   698→    }
   699→    out
   700→}

   755→/// Extract the last user-message content from an OpenAI-shaped `messages`
   756→/// array. Used as the retrieval query for Stage 2 (graph retrieval) and
   757→/// as the cache key for Stage 1 (action cache). Walks the messages in
   758→/// reverse so the *most recent* user turn wins (handles multi-turn chats
   759→/// where the user has refined their question across several messages).
   760→fn extract_user_query(body: &serde_json::Value) -> String {
   761→    let messages = match body.get("messages").and_then(|m| m.as_array()) {
   762→        Some(m) => m,
   763→        None => return String::new(),
   764→    };
   765→    for m in messages.iter().rev() {
   766→        if m.get("role").and_then(|r| r.as_str()) == Some("user") {
   767→            if let Some(c) = m.get("content") {
   768→                return content_to_string(c);
   769→            }
   770→        }
   771→    }
   772→    String::new()
   773→}
   774→
   775→/// Normalize a `content` field (which the OpenAI spec allows to be either
   776→/// a plain string or an array of `{type, text}` parts) into a single
   777→/// space-joined string. Non-text parts (images, tool calls, …) are
   778→/// dropped because the retrieval pipeline is text-only.
   779→fn content_to_string(c: &serde_json::Value) -> String {
   780→    match c {
   781→        serde_json::Value::String(s) => s.clone(),
   782→        serde_json::Value::Array(arr) => arr
   783→            .iter()
   784→            .filter_map(|p| {
   785→                if p.get("type").and_then(|t| t.as_str()) == Some("text") {
   786→                    p.get("text").and_then(|t| t.as_str()).map(|s| s.to_string())
   787→                } else {
   788→                    None
   789→                }
   790→            })
   791→            .collect::<Vec<_>>()
   792→            .join(" "),
   793→        _ => String::new(),
   794→    }
   795→}

   797→/// Extract the assistant's text content from an OpenAI-shaped chat
   798→/// completion response body. Returns `None` if the body isn't valid JSON
   799→/// or doesn't contain the expected `choices[0].message.content` path —
   800→/// the caller ([`call_backend`]) turns that into a fallback response.
   801→fn extract_assistant_content(body: &str) -> Option<String> {
   802→    let v: serde_json::Value = serde_json::from_str(body).ok()?;
   803→    v.get("choices")
   804→        .and_then(|c| c.get(0))
   805→        .and_then(|c| c.get("message"))
   806→        .and_then(|m| m.get("content"))
   807→        .and_then(|c| c.as_str())
   808→        .map(|s| s.to_string())
   809→}

   811→/// Build an OpenAI-shaped `chat.completion` JSON response from a plain
   812→/// text content string. Used by every code path that returns to the
   813→/// client (cache hit, pipeline output, fallback). The `model` field is

   828→        }],
   829→        "usage": { "prompt_tokens": 0, "completion_tokens": 0, "total_tokens": 0 }
   830→    })
   831→}

   849→/// Generate a pseudo-random hex ID from the current nanosecond timestamp.
   850→/// Used as the `id` field of synthetic `chat.completion` responses. Not
   851→/// cryptographically unique, but unique enough within a single process's
   852→/// lifetime to disambiguate concurrent responses.
   853→fn randomish_id() -> String {
   854→    use std::time::{SystemTime, UNIX_EPOCH};
   855→    let nanos = SystemTime::now()
   856→        .duration_since(UNIX_EPOCH)
   857→        .map(|d| d.as_nanos())
   858→        .unwrap_or(0);
   859→    format!("{:x}", nanos)
   860→}

   862→/// Current Unix timestamp in seconds. Used as the `created` field of
   863→/// synthetic `chat.completion` responses. Named `chrono_now` for historical
   864→/// reasons (a previous version used the `chrono` crate); the implementation
   865→/// now uses `std::time` directly to keep the dependency tree empty.
   866→fn chrono_now() -> u64 {
   867→    use std::time::{SystemTime, UNIX_EPOCH};
   868→    SystemTime::now()
   869→        .duration_since(UNIX_EPOCH)
   870→        .map(|d| d.as_secs())
   871→        .unwrap_or(0)
   872→}

   874→/// Speculative prefetch: warm the retrieval cache for graph-adjacent candidate
   875→/// queries so the next, related query is instant.
   876→async fn prefetch(state: AppState, retrieved: &[ScoredNode]) {
   877→    if retrieved.is_empty() {
   878→        return;
   879→    }
   880→    let graph = state.graph.clone();
   881→    let cache = state.cache.clone();
   882→    let candidates: Vec<String> = retrieved.iter().take(3).map(|n| n.text.clone()).collect();
   883→    tokio::spawn(async move {
   884→        for q in candidates {
   885→            let (qvec, ctx) = {
   886→                let g = graph.lock().await;
   887→                let qv = g.vectorizer.vectorize(&q);
   888→                let r = g.retrieve(&q, 8);
   889→                let c = compress::compress(&r, &q, 6000);
   890→                (qv, c)
   891→            };
   892→            cache.lock().await.put_retrieval(&q, qvec, ctx);
   893→        }
   894→    });
   895→}

   909→/// OpenAI-compatible chat-completions **streaming** endpoint.
   910→///
   911→/// Accepts the same body shape as [`chat_completions`] (`{ messages: [...] }`)
   912→/// and returns a `text/event-stream` response. Each token of the final
   913→/// assistant message is emitted as an SSE event of the form:
   914→///
   915→/// ```text
   916→/// event: token
   917→/// data: {"token": "hello"}
   918→///
   919→/// ```
   920→///
   921→/// A terminal `event: done` event signals end-of-stream.
   922→///
   923→/// # Implementation note
   924→///
   925→/// Aether Engine's backend HTTP client is not natively streaming (it would
   926→/// require a chunked-transfer-aware `reqwest::Body` adapter), so this handler
   927→/// runs the simplified pipeline (graph retrieval → compression → one
   928→/// `call_backend` round-trip) and then **emits the final response token by
   929→/// token** as SSE events. The browser's `EventSource` parser sees one event
   930→/// per token; the only difference from a true token-stream is that all events
   931→/// arrive in a single HTTP body rather than being interleaved with backend
   932→/// generation. This is the fallback shape explicitly allowed by the spec
   933→/// ("simple chunked HTTP response with `text/event-stream` content type").
   934→pub async fn chat_completions_stream(
   935→    State(state): State<AppState>,
   936→    Json(body): Json<serde_json::Value>,
   937→) -> impl IntoResponse {
   938→    state.stats.lock().await.requests += 1;
   939→    let query = extract_user_query(&body);
   940→
   941→    // Stage 2: graph retrieval (TF-IDF + edge expansion).
   942→    let retrieved: Vec<ScoredNode> = {
   943→        let g = state.graph.lock().await;
   944→        g.retrieve(&query, 8)
   945→    };
   946→    // Stage 3: context compression.
   947→    let context_block = compress::compress(&retrieved, &query, 6000);
   948→
   949→    // Simplified pipeline: one backend call. The full chat_completions
   950→    // pipeline (decomposition / ATD / distillation) is intentionally not
   951→    // duplicated here — the streaming endpoint is for live UX, the regular
   952→    // endpoint is for highest-quality responses.
   953→    let response = call_backend(&state, &body, &context_block).await;
   954→
   955→    // Build the SSE body: one `event: token` per whitespace-split token of
   956→    // the final response, terminated by `event: done`.
   957→    let mut sse = String::with_capacity(response.len().saturating_mul(2));
   958→    for token in response.split_whitespace() {
   959→        // JSON-encode the token (handles embedded quotes / backslashes) and
   960→        // interpolate it into the `data:` line. The leading space after
   961→        // `data:` is required by the SSE spec.
   962→        let token_json = serde_json::to_string(token).unwrap_or_else(|_| "\"\"".to_string());
   963→        sse.push_str(&format!(
   964→            "event: token\ndata: {{\"token\": {}}}\n\n",
   965→            token_json
   966→        ));
   967→    }
   968→    sse.push_str("event: done\ndata: {\"done\": true}\n\n");
   969→
   970→    (
   971→        axum::http::StatusCode::OK,
   972→        [("content-type", "text/event-stream")],
   973→        sse,
   974→    )
   975→}

   998→        let h = state.hcm.lock().await;
   999→        (h.pair_count, h.capacity(), h.memory_bytes())
  1000→    };
  1001→
  1002→    let data = crate::dashboard::DashboardData {
  1003→        engine_name: "Aether Engine",
  1004→        version: "3.0",
  1005→        uptime_seconds: crate::dashboard::uptime_seconds(),
  1006→        node_count: nodes,
  1007→        edge_count: edges,
  1008→        cache_size,
  1009→        decompositions,
  1010→        atd_verdicts,
  1011→        hcm_state_bytes: hcm_memory,
  1012→        hcm_pairs,
  1013→        hcm_capacity,
  1014→        requests,

  1022→// GET /metrics — Prometheus exposition format
  1023→// ---------------------------------------------------------------------------
  1024→
  1025→/// Prometheus-format metrics endpoint.
  1026→///
  1027→/// Returns a `text/plain; version=0.0.4` body in Prometheus exposition
  1028→/// format. Five series are exposed:
  1029→///
  1030→/// - `aether_graph_nodes` (gauge) — nodes in the semantic memory graph.
  1031→/// - `aether_graph_edges` (gauge) — directed edges in the adjacency list.
  1032→/// - `aether_cache_size` (gauge) — action-cache entries.
  1033→/// - `aether_decompositions_total` (counter) — cognitive decompositions.
  1034→/// - `aether_atd_verdicts_total` (counter) — ATD verifications performed.
  1035→///
  1036→/// Designed to be scraped by a Prometheus / Grafana agent with no extra
  1037→/// configuration.
  1038→pub async fn prometheus_metrics(State(state): State<AppState>) -> impl IntoResponse {
  1039→    let (nodes, edges) = {
  1040→        let g = state.graph.lock().await;
  1041→        (g.len(), g.edge_count())
  1042→    };
  1043→    let cache_size = state.cache.lock().await.len();
  1044→    let (decompositions, atd_verifications) = {
  1045→        let s = state.stats.lock().await;
  1046→        (s.decompositions, s.atd_verifications)
  1047→    };
  1048→
  1049→    // Build the Prometheus exposition body line-by-line. Each line MUST NOT
  1050→    // have leading whitespace (the exposition format is line-oriented and
  1051→    // strict about data lines starting with the metric name), so we use
  1052→    // `push_str` rather than a multi-line `format!` with `\` continuations
  1053→    // (which would embed the source-file indentation into the string).
  1054→    let mut body = String::with_capacity(1024);
  1055→    body.push_str("# HELP aether_graph_nodes Total nodes in memory graph\n");
  1056→    body.push_str("# TYPE aether_graph_nodes gauge\n");
  1057→    body.push_str(&format!("aether_graph_nodes {}\n", nodes));
  1058→    body.push_str("# HELP aether_graph_edges Total edges in memory graph\n");
  1059→    body.push_str("# TYPE aether_graph_edges gauge\n");
  1060→    body.push_str(&format!("aether_graph_edges {}\n", edges));
  1061→    body.push_str("# HELP aether_cache_size Action cache entries\n");
  1062→    body.push_str("# TYPE aether_cache_size gauge\n");
  1063→    body.push_str(&format!("aether_cache_size {}\n", cache_size));
  1064→    body.push_str("# HELP aether_decompositions_total Total cognitive decompositions\n");
  1065→    body.push_str("# TYPE aether_decompositions_total counter\n");
  1066→    body.push_str(&format!("aether_decompositions_total {}\n", decompositions));
  1067→    body.push_str("# HELP aether_atd_verdicts_total Total ATD verdicts\n");
  1068→    body.push_str("# TYPE aether_atd_verdicts_total counter\n");
  1069→    body.push_str(&format!("aether_atd_verdicts_total {}\n", atd_verifications));
  1070→
  1071→    (
  1072→        axum::http::StatusCode::OK,
  1073→        [("content-type", "text/plain; version=0.0.4")],
  1074→        body,
  1075→    )
  1076→}

  1119→// POST /graph/import — bulk-add nodes from a JSON array
  1120→// ---------------------------------------------------------------------------
  1121→
  1122→/// Bulk-import nodes into the semantic memory graph.
  1123→///
  1124→/// Accepts a JSON body of the shape `{ "nodes": [ { "id": "...", "text": "..." }, ... ] }`.
  1125→/// Each entry is fed through [`MemoryGraph::add`](crate::graph::MemoryGraph::add)
  1126→/// (the same path used by `POST /graph/add`), with `kind` and `metadata`
  1127→/// defaulting to `"fact"` and `{}` respectively when absent.
  1128→///
  1129→/// Returns `{ "imported": <count> }` where `count` is the number of nodes
  1130→/// successfully added. Entries missing `id` or `text` are skipped.
  1131→pub async fn graph_import(
  1132→    State(state): State<AppState>,
  1133→    Json(body): Json<serde_json::Value>,
  1134→) -> impl IntoResponse {
  1135→    let mut imported: usize = 0;
  1136→    if let Some(nodes) = body.get("nodes").and_then(|n| n.as_array()) {
  1137→        let mut g = state.graph.lock().await;
  1138→        for node in nodes {
  1139→            let id = node.get("id").and_then(|v| v.as_str()).unwrap_or("").to_string();
  1140→            let text = node.get("text").and_then(|v| v.as_str()).unwrap_or("").to_string();
  1141→            if id.is_empty() || text.is_empty() {
  1142→                continue;
  1143→            }
  1144→            let kind = node
  1145→                .get("kind")
  1146→                .and_then(|v| v.as_str())
  1147→                .unwrap_or("fact")
  1148→                .to_string();
  1149→            let metadata = node.get("metadata").cloned().unwrap_or(json!({}));
  1150→            g.add(AddNodeRequest {
  1151→                id,
  1152→                text,
  1153→                kind,
  1154→                metadata,
  1155→            });
  1156→            imported = imported.saturating_add(1);
  1157→        }
  1158→    }
  1159→    Json(json!({ "imported": imported }))
  1160→}

  1220→// ---------------------------------------------------------------------------
  1221→// GET /graph/stats — detailed graph statistics
  1222→// ---------------------------------------------------------------------------
  1223→
  1224→/// Detailed statistics about the semantic memory graph.
  1225→///
  1226→/// Returns:
  1227→///
  1228→/// - `node_count` — total nodes.
  1229→/// - `edge_count` — total directed edges in the adjacency list (each
  1230→///   undirected edge is stored twice, once per endpoint).
  1231→/// - `avg_edges_per_node` — `edge_count / node_count` (0 if empty).
  1232→/// - `max_edges` — the largest adjacency list of any single node (capped at
  1233→///   `top_k = 5`).
  1234→/// - `density` — `edge_count / (N * (N-1))` for `N > 1`, else 0 (the
  1235→///   directed-edge density of the adjacency list).
  1236→/// - `top_connected_nodes` — the top 5 nodes by adjacency-list length, each
  1237→///   as `{ id, text, edge_count }`.
  1238→pub async fn graph_stats(State(state): State<AppState>) -> impl IntoResponse {
  1239→    let g = state.graph.lock().await;
  1240→    let node_count = g.len();
  1241→    let edge_count = g.edge_count();
  1242→    let max_edges = g
  1243→        .adjacency
  1244→        .values()
  1245→        .map(|v| v.len())
  1246→        .max()
  1247→        .unwrap_or(0);
  1248→    let avg_edges_per_node = if node_count > 0 {
  1249→        edge_count as f64 / node_count as f64
  1250→    } else {
  1251→        0.0
  1252→    };
  1253→    let density = if node_count > 1 {
  1254→        edge_count as f64 / (node_count as f64 * (node_count as f64 - 1.0))
  1255→    } else {
  1256→        0.0
  1257→    };
  1258→
  1259→    // Top connected nodes: sort adjacency entries by length (desc), take 5.
  1260→    // We clone the (id, edge_count) pairs out of the adjacency map so we can
  1261→    // drop the borrow before looking up node texts (avoiding a double-borrow
  1262→    // of `g.adjacency` and `g.nodes`).
  1263→    let mut node_edge_counts: Vec<(String, usize)> = g
  1264→        .adjacency
  1265→        .iter()
  1266→        .map(|(id, nbrs)| (id.clone(), nbrs.len()))
  1267→        .collect();
  1268→    node_edge_counts.sort_by(|a, b| b.1.cmp(&a.1));
  1269→
  1270→    let top_connected: Vec<serde_json::Value> = node_edge_counts
  1271→        .into_iter()
  1272→        .take(5)
  1273→        .filter_map(|(id, ec)| {
  1274→            g.nodes.get(&id).map(|n| {
  1275→                json!({
  1276→                    "id": id,
  1277→                    "text": n.text.clone(),
  1278→                    "edge_count": ec,
  1279→                })
  1280→            })
  1281→        })
  1282→        .collect();
  1283→
  1284→    Json(json!({
  1285→        "node_count": node_count,
  1286→        "edge_count": edge_count,
  1287→        "avg_edges_per_node": avg_edges_per_node,
  1288→        "max_edges": max_edges,
  1289→        "density": density,
  1290→        "top_connected_nodes": top_connected,
  1291→    }))
  1292→}

  1302→// ---------------------------------------------------------------------------
  1303→// POST /v1/agent/run — run the autonomous agent loop
  1304→// ---------------------------------------------------------------------------
  1305→
  1306→/// Run the autonomous agent loop for a single goal.
  1307→///
  1308→/// Accepts a JSON body of the shape:
  1309→///
  1310→/// ```json
  1311→/// {
  1312→///   "goal": "set up a Python dev environment and write a hello-world script",
  1313→///   "context": { "windows": [...], "memory": [...], "plan": {...} },
  1314→///   "max_iterations": 20
  1315→/// }
  1316→/// ```
  1317→///
  1318→/// - `goal` (required) — natural-language goal the agent should achieve.
  1319→/// - `context` (optional, default `{}`) — current OS state, pretty-printed
  1320→///   into the system prompt so the LLM perceives open windows, memory,
  1321→///   active plan, etc.
  1322→/// - `max_iterations` (optional, default `20`, clamped to `[1, 50]`) —
  1323→///   hard cap on think iterations.
  1324→///
  1325→/// Returns:
  1326→///
  1327→/// ```json
  1328→/// {
  1329→///   "ok": true,
  1330→///   "result": "<final assistant text>",
  1331→///   "iterations": 3,
  1332→///   "completed": true,
  1333→///   "tool_calls": [
  1334→///     { "name": "file_read", "params": { "path": "/etc/hostname" } },
  1335→///     ...
  1336→///   ]
  1337→/// }
  1338→/// ```
  1339→///
  1340→/// The `tool_calls` array is the full transcript of every tool call the
  1341→/// LLM made across all iterations, in execution order. The Next.js layer
  1342→/// uses this transcript to perform the real OS side-effects (file IO,
  1343→/// PTY exec, window manager mutations, …) — the Aether Engine's tool
  1344→/// executors return placeholders by design (see [`crate::tools`]).
  1345→///
  1346→/// On a missing `goal` field the handler returns
  1347→/// `{ ok: false, error: "missing 'goal' field" }` without starting the
  1348→/// loop.
  1349→pub async fn agent_run(
  1350→    State(state): State<AppState>,
  1351→    Json(body): Json<serde_json::Value>,
  1352→) -> impl IntoResponse {
  1353→    let goal = body
  1354→        .get("goal")
  1355→        .and_then(|g| g.as_str())
  1356→        .unwrap_or("")
  1357→        .to_string();
  1358→    let context = body.get("context").cloned().unwrap_or(json!({}));
  1359→    let max_iterations = body
  1360→        .get("max_iterations")
  1361→        .and_then(|m| m.as_u64())
  1362→        .map(|n| n as usize)
  1363→        .unwrap_or(crate::agent::DEFAULT_MAX_ITERATIONS);
  1364→
  1365→    if goal.is_empty() {
  1366→        return Json(json!({
  1367→            "ok": false,
  1368→            "error": "missing 'goal' field",
  1369→        }))
  1370→        .into_response();
  1371→    }
  1372→
  1373→    let result = crate::agent::run_agent_loop(state, &goal, &context, max_iterations).await;
  1374→
  1375→    Json(json!({
  1376→        "ok": true,
  1377→        "result": result.result,
  1378→        "iterations": result.iterations,
  1379→        "completed": result.completed,
  1380→        "tool_calls": result.tool_calls.iter().map(|c| json!({
  1381→            "name": c.name,
  1382→            "params": c.params,
  1383→        })).collect::<Vec<_>>(),
  1384→    }))
  1385→    .into_response()
  1386→}
