     1→//! # Tools — OS-control tool definitions for the Aether agentic layer.
     2→//!
     3→//! This module defines the **tool surface** the autonomous agent can use to
     4→//! perceive and manage a Linux-like OS. Each tool is:
     5→//!
     6→//! - a [`Tool`] enum variant (carries the parsed parameters),
     7→//! - a [`ToolSpec`] entry in the [`ToolRegistry`] catalog (carries the
     8→//!   name + description + JSON-Schema parameters used to advertise the tool
     9→//!   to the LLM and to the `GET /v1/tools` introspection endpoint),
    10→//! - a dispatch arm in [`ToolRegistry::execute`] (carries the placeholder
    11→//!   executor).
    12→//!
    13→//! # Why placeholders?
    14→//!
    15→//! The Aether Engine runs as a Rust HTTP service; it does not itself touch
    16→//! the filesystem, spawn shells, or open GUI windows. The actual OS
    17→//! integration happens through the Next.js layer in Alpha-OS, which:
    18→//!
    19→//! 1. Calls `POST /v1/agent/run` to start the agentic loop,
    20→//! 2. Receives the parsed [`ToolCall`](crate::agent::ToolCall)s back as JSON,
    21→//! 3. Performs the real OS action (file IO, PTY exec, window manager
    22→//!    mutation, graph memory add, …),
    23→//! 4. Feeds the resulting observation back into the next agent iteration via
    24→//!    the `context` field of the next `POST /v1/agent/run` call.
    25→//!
    26→//! The placeholder executors here return **canonical "what would happen"
    27→//! responses** so the agent loop can be tested end-to-end (perceive → think
    28→//! → act → observe) before the OS wiring is complete, and so the structure
    29→//! of the tool registry is exercised by every iteration even when no real
    30→//! side-effects are available.
    31→//!
    32→//! # Adding a new tool
    33→//!
    34→//! 1. Add a variant to [`Tool`] carrying its parsed parameters.
    35→//! 2. Add a `(name, description, schema)` triple to
    36→//!    [`ToolRegistry::catalog`].
    37→//! 3. Add a dispatch arm to [`ToolRegistry::execute`] returning a
    38→//!    placeholder result.
    39→//! 4. Add a constructor arm to [`Tool::from_request`].
    40→
    41→use serde_json::{json, Value};
    42→
    43→// ---------------------------------------------------------------------------
    44→// Tool enum — one variant per available tool, carrying parsed params
    45→// ---------------------------------------------------------------------------
    46→
    47→/// A parsed tool call — the variant identifies *which* tool, and the fields
    48→/// carry the validated parameters extracted from the LLM's JSON payload by
    49→/// [`Tool::from_request`].
    50→///
    51→/// Variants are constructed only by [`Tool::from_request`]; the agent loop
    52→/// never builds them directly. Unknown tool names or missing required
    53→/// parameters yield `None`, which the agent loop surfaces as a
    54→/// `tool_not_found` error message back to the LLM.
    55→#[derive(Clone, Debug)]
    56→pub enum Tool {
    57→    /// Read a file from disk. Returns the file contents as a UTF-8 string.
    58→    FileRead {
    59→        /// Absolute or relative path of the file to read.
    60→        path: String,
    61→    },
    62→    /// Write `content` to `path`, overwriting if the file already exists and
    63→    /// creating it (including parent directories) if it does not.
    64→    FileWrite {
    65→        /// Path of the file to write.
    66→        path: String,
    67→        /// The full text content to write.
    68→        content: String,
    69→    },
    70→    /// List the entries of a directory. Returns `{ name, is_dir }` per entry.
    71→    FileList {
    72→        /// Path of the directory to list.
    73→        path: String,
    74→    },
    75→    /// Delete a file or directory. Directories are removed recursively.
    76→    FileDelete {
    77→        /// Path of the file or directory to delete.
    78→        path: String,
    79→    },
    80→    /// Run a shell command in the user's default shell. Returns
    81→    /// `{ stdout, stderr, exit_code }`.
    82→    Exec {
    83→        /// The full shell command line (interpreted by `/bin/bash -c`).
    84→        command: String,
    85→    },
    86→    /// Open an application window in the Alpha-OS desktop.
    87→    WindowOpen {
    88→        /// App identifier (e.g. `"terminal"`, `"editor"`, `"browser"`).
    89→        app: String,
    90→    },
    91→    /// Close an open application window by its id.
    92→    WindowClose {
    93→        /// The window id returned by `window_open` or listed in the OS state.
    94→        window_id: String,
    95→    },
    96→    /// Add a memory to Akasha (the semantic memory graph). The memory
    97→    /// becomes a new node and is automatically linked to its semantic
    98→    /// neighbors.
    99→    MemoryAdd {
   100→        /// The memory text.
   101→        text: String,
   102→        /// Memory kind: `"fact"`, `"lesson"`, `"plan"`, `"goal"`,
   103→        /// `"intention"`, `"log"`, or `"code"`.
   104→        kind: String,
   105→    },
   106→    /// Search the semantic memory graph for memories matching `query`.
   107→    /// Returns the top results with cosine-similarity scores.
   108→    MemorySearch {
   109→        /// The free-text query to search for.
   110→        query: String,
   111→    },
   112→    /// Search the public web for `query`. Returns a list of result snippets
   113→    /// with URLs.
   114→    WebSearch {
   115→        /// The search query.
   116→        query: String,
   117→    },
   118→    /// Create a multi-step plan. The plan is stored and surfaced back to the
   119→    /// agent on subsequent iterations as part of the OS state, so the agent
   120→    /// can track its progress through the steps.
   121→    PlanCreate {
   122→        /// The high-level goal the plan achieves.
   123→        goal: String,
   124→        /// The ordered list of step descriptions.
   125→        steps: Vec<String>,
   126→    },
   127→    /// Mark a step in the current plan as done. Steps are 0-indexed.
   128→    PlanUpdate {
   129→        /// The index of the step to mark as done.
   130→        step_index: usize,
   131→    },
   132→}
   133→
   134→impl Tool {
   135→    /// Canonical name the LLM uses to reference this tool. Mirrors the
   136→    /// `name` field in the corresponding [`ToolSpec`].
   137→    pub fn name(&self) -> &'static str {
   138→        match self {
   139→            Tool::FileRead { .. } => "file_read",
   140→            Tool::FileWrite { .. } => "file_write",
   141→            Tool::FileList { .. } => "file_list",
   142→            Tool::FileDelete { .. } => "file_delete",
   143→            Tool::Exec { .. } => "exec",
   144→            Tool::WindowOpen { .. } => "window_open",
   145→            Tool::WindowClose { .. } => "window_close",
   146→            Tool::MemoryAdd { .. } => "memory_add",
   147→            Tool::MemorySearch { .. } => "memory_search",
   148→            Tool::WebSearch { .. } => "web_search",
   149→            Tool::PlanCreate { .. } => "plan_create",
   150→            Tool::PlanUpdate { .. } => "plan_update",
   151→        }
   152→    }
   153→
   154→    /// Human-readable description shown to the LLM in the system prompt and
   155→    /// to clients via `GET /v1/tools`. Mirrors the `description` field in
   156→    /// the corresponding [`ToolSpec`].
   157→    pub fn description(&self) -> &'static str {
   158→        match self {
   159→            Tool::FileRead { .. } => "Read the full contents of a file from disk.",
   160→            Tool::FileWrite { .. } => {
   161→                "Write text content to a file, overwriting if present and creating \
   162→                 parent directories as needed."
   163→            }
   164→            Tool::FileList { .. } => "List the entries of a directory.",
   165→            Tool::FileDelete { .. } => {
   166→                "Delete a file or directory (directories are removed recursively)."
   167→            }
   168→            Tool::Exec { .. } => {
   169→                "Run a shell command in the user's default shell and return \
   170→                 stdout, stderr, and exit code."
   171→            }
   172→            Tool::WindowOpen { .. } => "Open an application window on the OS desktop.",
   173→            Tool::WindowClose { .. } => "Close an open application window by id.",
   174→            Tool::MemoryAdd { .. } => {
   175→                "Add a memory (fact/lesson/plan/goal/intention/log/code) to \
   176→                 the Akasha semantic memory graph."
   177→            }
   178→            Tool::MemorySearch { .. } => {
   179→                "Search the semantic memory graph for matching memories, \
   180→                 ranked by cosine similarity."
   181→            }
   182→            Tool::WebSearch { .. } => "Search the public web for fresh information.",
   183→            Tool::PlanCreate { .. } => {
   184→                "Create a multi-step plan for a goal. The plan is persisted \
   185→                 and surfaced back to the agent on subsequent iterations."
   186→            }
   187→            Tool::PlanUpdate { .. } => "Mark a step (0-indexed) in the current plan as done.",
   188→        }
   189→    }
   190→
   191→    /// Construct a [`Tool`] from a `(name, params)` pair extracted from the
   192→    /// LLM's tool-call payload by
   193→    /// [`parse_tool_calls`](crate::agent::parse_tool_calls).
   194→    ///
   195→    /// Returns `None` when:
   196→    /// - the tool name is unknown to the registry,
   197→    /// - a required parameter is missing,
   198→    /// - a parameter has the wrong JSON type.
   199→    ///
   200→    /// The agent loop converts `None` into a `tool_not_found` /
   201→    /// `invalid_params` tool-result message so the LLM can self-correct on
   202→    /// the next iteration.
   203→    pub fn from_request(name: &str, params: &Value) -> Option<Self> {
   204→        match name {
   205→            "file_read" => {
   206→                let path = params.get("path")?.as_str()?.to_string();
   207→                Some(Tool::FileRead { path })
   208→            }
   209→            "file_write" => {
   210→                let path = params.get("path")?.as_str()?.to_string();
   211→                let content = params
   212→                    .get("content")
   213→                    .and_then(|c| c.as_str())
   214→                    .unwrap_or("")
   215→                    .to_string();
   216→                Some(Tool::FileWrite { path, content })
   217→            }
   218→            "file_list" => {
   219→                let path = params.get("path")?.as_str()?.to_string();
   220→                Some(Tool::FileList { path })
   221→            }
   222→            "file_delete" => {
   223→                let path = params.get("path")?.as_str()?.to_string();
   224→                Some(Tool::FileDelete { path })
   225→            }
   226→            "exec" => {
   227→                let command = params.get("command")?.as_str()?.to_string();
   228→                Some(Tool::Exec { command })
   229→            }
   230→            "window_open" => {
   231→                let app = params.get("app")?.as_str()?.to_string();
   232→                Some(Tool::WindowOpen { app })
   233→            }
   234→            "window_close" => {
   235→                let window_id = params.get("window_id")?.as_str()?.to_string();
   236→                Some(Tool::WindowClose { window_id })
   237→            }
   238→            "memory_add" => {
   239→                let text = params.get("text")?.as_str()?.to_string();
   240→                let kind = params
   241→                    .get("kind")
   242→                    .and_then(|k| k.as_str())
   243→                    .unwrap_or("fact")
   244→                    .to_string();
   245→                Some(Tool::MemoryAdd { text, kind })
   246→            }
   247→            "memory_search" => {
   248→                let query = params.get("query")?.as_str()?.to_string();
   249→                Some(Tool::MemorySearch { query })
   250→            }
   251→            "web_search" => {
   252→                let query = params.get("query")?.as_str()?.to_string();
   253→                Some(Tool::WebSearch { query })
   254→            }
   255→            "plan_create" => {
   256→                let goal = params.get("goal")?.as_str()?.to_string();
   257→                let steps = params
   258→                    .get("steps")
   259→                    .and_then(|s| s.as_array())
   260→                    .map(|arr| {
   261→                        arr.iter()
   262→                            .filter_map(|s| s.as_str().map(|s| s.to_string()))
   263→                            .collect::<Vec<_>>()
   264→                    })
   265→                    .unwrap_or_default();
   266→                Some(Tool::PlanCreate { goal, steps })
   267→            }
   268→            "plan_update" => {
   269→                let step_index = params.get("step_index")?.as_u64()? as usize;
   270→                Some(Tool::PlanUpdate { step_index })
   271→            }
   272→            _ => None,
   273→        }
   274→    }
   275→}

   277→// ---------------------------------------------------------------------------
   278→// ToolSpec — name + description + JSON-Schema parameters
   279→// ---------------------------------------------------------------------------
   280→
   281→/// A tool advertisement: name, description, and a JSON Schema describing
   282→/// the tool's parameters. Returned by [`ToolRegistry::catalog`] for
   283→/// injection into the agent system prompt and for the `GET /v1/tools`
   284→/// introspection endpoint.
   285→#[derive(Clone, Debug)]
   286→pub struct ToolSpec {
   287→    /// Canonical tool name (matches [`Tool::name`]).
   288→    pub name: &'static str,
   289→    /// Human-readable description (matches [`Tool::description`]).
   290→    pub description: &'static str,
   291→    /// JSON Schema (draft-07 subset) describing the tool's parameters.
   292→    /// `type: "object"` with `properties` and `required` arrays.
   293→    pub parameters: Value,
   294→}

   296→// ---------------------------------------------------------------------------
   297→// ToolRegistry — catalog + dispatch
   298→// ---------------------------------------------------------------------------
   299→
   300→/// Stateless registry of all tools available to the autonomous agent.
   301→///
   302→/// Despite having no fields, this is a struct (not free functions) so that
   303→/// future evolution (per-agent tool whitelisting, dynamic tool registration,
   304→/// rate limits, capability scopes, …) has an obvious extension point. Today
   305→/// every call to [`ToolRegistry::new`] returns an identical catalog.
   306→///
   307→/// # Example
   308→///
   309→/// ```ignore
   310→/// use crate::tools::{Tool, ToolRegistry};
   311→///
   312→/// let registry = ToolRegistry::new();
   313→/// let catalog = registry.catalog();
   314→/// assert_eq!(catalog.len(), 12);
   315→///
   316→/// let params = serde_json::json!({ "path": "/etc/hostname" });
   317→/// let tool = Tool::from_request("file_read", &params).unwrap();
   318→/// let (output, success) = futures::executor::block_on(registry.execute(&tool));
   319→/// assert!(success);
   320→/// ```
   321→pub struct ToolRegistry;
   322→

   527→    /// Execute a parsed [`Tool`].
   528→    ///
   529→    /// Returns `(output, success)` where:
   530→    /// - `output` is a human/LLM-readable string describing the result,
   531→    /// - `success` is `true` when the tool "succeeded" (placeholder logic;
   532→    ///   see the module docs for why real OS integration happens via the
   533→    ///   Next.js layer).
   534→    ///
   535→    /// Placeholder behavior:
   536→    ///
   537→    /// - File/shell/window/web tools always succeed and return a canonical
   538→    ///   `[<tool> placeholder] would <action>: <params>` string. The Next.js
   539→    ///   layer intercepts these and performs the real side-effect.
   540→    /// - Memory tools (`memory_add`, `memory_search`) and plan tools
   541→    ///   (`plan_create`, `plan_update`) also return placeholders — the
   542→    ///   Next.js layer is responsible for POSTing to `/graph/add` /
   543→    ///   `/graph/search` and updating the plan store respectively.
   544→    ///
   545→    /// This split keeps the tool registry self-contained (no `AppState`
   546→    /// dependency, no async I/O for files, no PTY plumbing) while still
   547→    /// exercising the full agent loop end-to-end.
   548→    pub async fn execute(&self, tool: &Tool) -> (String, bool) {
   549→        match tool {
   550→            Tool::FileRead { path } => (
   551→                format!(
   552→                    "[file_read placeholder] would read file at \"{path}\" and return its contents."
   553→                ),
   554→                true,
   555→            ),
   556→            Tool::FileWrite { path, content } => (
   557→                format!(
   558→                    "[file_write placeholder] would write {} byte(s) to \"{path}\".",
   559→                    content.len()
   560→                ),
   561→                true,
   562→            ),
   563→            Tool::FileList { path } => (
   564→                format!(
   565→                    "[file_list placeholder] would list the directory entries of \"{path}\"."
   566→                ),
   567→                true,
   568→            ),
   569→            Tool::FileDelete { path } => (
   570→                format!(
   571→                    "[file_delete placeholder] would delete the file or directory at \"{path}\"."
   572→                ),
   573→                true,
   574→            ),
   575→            Tool::Exec { command } => (
   576→                format!(
   577→                    "[exec placeholder] would run shell command: `{command}`"
   578→                ),
   579→                true,
   580→            ),
   581→            Tool::WindowOpen { app } => (
   582→                format!(
   583→                    "[window_open placeholder] would open the \"{app}\" application window."
   584→                ),
   585→                true,
   586→            ),
   587→            Tool::WindowClose { window_id } => (
   588→                format!(
   589→                    "[window_close placeholder] would close window \"{window_id}\"."
   590→                ),
   591→                true,
   592→            ),
   593→            Tool::MemoryAdd { text, kind } => (
   594→                format!(
   595→                    "[memory_add placeholder] would add a \"{kind}\" memory to Akasha: {}",
   596→                    truncate(text, 200)
   597→                ),
   598→                true,
   599→            ),
   600→            Tool::MemorySearch { query } => (
   601→                format!(
   602→                    "[memory_search placeholder] would search the semantic memory graph for: {}",
   603→                    truncate(query, 200)
   604→                ),
   605→                true,
   606→            ),
   607→            Tool::WebSearch { query } => (
   608→                format!(
   609→                    "[web_search placeholder] would search the web for: {}",
   610→                    truncate(query, 200)
   611→                ),
   612→                true,
   613→            ),
   614→            Tool::PlanCreate { goal, steps } => (
   615→                format!(
   616→                    "[plan_create placeholder] would create a {}-step plan for goal: {}\n  - {}",
   617→                    steps.len(),
   618→                    truncate(goal, 200),
   619→                    steps
   620→                        .iter()
   621→                        .map(|s| truncate(s, 100))
   622→                        .collect::<Vec<_>>()
   623→                        .join("\n  - ")
   624→                ),
   625→                true,
   626→            ),
   627→            Tool::PlanUpdate { step_index } => (
   628→                format!(
   629→                    "[plan_update placeholder] would mark plan step #{step_index} as done."
   630→                ),
   631→                true,
   632→            ),
   633→        }
   634→    }
   635→}

   648→/// Truncate `s` to at most `max` characters, appending an ellipsis when
   649→/// truncation occurs. Used to keep placeholder tool outputs short enough
   650→/// to fit comfortably inside the LLM's context window across many
   651→/// iterations.
   652→fn truncate(s: &str, max: usize) -> String {
   653→    if s.chars().count() <= max {
   654→        s.to_string()
   655→    } else {
   656→        let truncated: String = s.chars().take(max).collect();
   657→        format!("{truncated}…")
   658→    }
   659→}
