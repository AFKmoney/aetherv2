     1→     1→//! # HTTP Handlers — the 10-stage Cognitive Pipeline (Aether Engine v3.0)
     2→     2→//!
     3→     3→//! All Axum route handlers live here. The flagship handler,
     4→     4→//! [`chat_completions`], implements the full cognitive pipeline that gives
     5→     5→//! Aether Engine its 10x capacity multiplier over a vanilla GGUF backend:
     6→     6→//!
     7→     7→//!   1. **Action cache check** — instant for repeated/similar queries
     8→     8→//!      ([`crate::cache::ActionCache`]).
     9→     9→//!   2. **Graph retrieval** — TF-IDF semantic search + 1-hop edge expansion
    10→    10→//!      ([`crate::graph::MemoryGraph::retrieve`]).
    11→    11→//!   3. **Context compression** — 40K→4K tokens preserving signal
    12→    12→//!      ([`crate::compress::compress`]).
    13→    13→//!   4. **Complexity analysis** — Simple / Moderate / Complex
    14→    14→//!      ([`crate::decompose::analyze_complexity`]).
    15→    15→//!   5. **Cognitive decomposition** — Complex queries are broken into
    16→    16→//!      sub-questions ([`crate::decompose::decompose`]).
    17→    17→//!   6. **Sequential sub-question solving** — each with fresh context +
    18→    18→//!      dependency injection (driven by this module via [`call_backend`]).
    19→    19→//!   7. **Synthesis** — combine sub-answers into final response (the
    20→    20→//!      `"synth"` sub-question, or a plain concatenation fallback).
    21→    21→//!   8. **ATD verification** — dual-graph likelihood-vs-entropy collision;
    22→    22→//!      retry once if failed ([`crate::atd::verify`]).
    23→    23→//!   9. **Knowledge distillation** — store successful decomposition patterns
    24→    24→//!      ([`crate::decompose::DistillationStore`]).
    25→    25→//!  10. **Speculative prefetch** — warm the retrieval cache for
    26→    26→//!      graph-adjacent queries so the next related query is instant
    27→    27→//!      ([`prefetch`]).
    28→    28→//!
    29→    29→//! # OpenAI compatibility
    30→    30→//!
    31→    31→//! [`chat_completions`] mirrors the OpenAI `POST /v1/chat/completions`
    32→    32→//! contract (request body in, chat-completion-shaped JSON out) so the
    33→    33→//! engine can be dropped in as a drop-in replacement for any OpenAI-compatible
    34→    34→//! client. The backend it forwards to is *also* OpenAI-compatible (default
    35→    35→//! `http://localhost:11434/v1`, i.e. Ollama).

   164→   164→// POST /graph/add
   165→   165→// ---------------------------------------------------------------------------
   166→   166→
   167→   167→/// Add (or replace) a node in the semantic memory graph.
   168→   168→///
   169→   169→/// Side-effect: the new node is *also* folded into the Holographic Context
   170→   170→/// Memory arena as `(hash(id), hash(text))` so the HCM has a parallel
   171→   171→/// associative copy of every graph memory. The fold counter in
   172→   172→/// [`Stats::hcm_pairs_folded`](crate::Stats) is bumped accordingly.
   173→   173→///
   174→   174→/// Returns `{ ok, id, edges_created }` where `edges_created` is the total
   175→   175→/// number of adjacency entries created across the whole graph by the
   176→   176→/// recompute (see [`MemoryGraph::add`](crate::graph::MemoryGraph::add)).
   177→   177→pub async fn graph_add(
   178→   178→    State(state): State<AppState>,
   179→   179→    Json(req): Json<AddNodeRequest>,
   180→   180→) -> impl IntoResponse {
   181→   181→    let id = req.id.clone();
   182→   182→    let text = req.text.clone();
   183→   183→    let edges_created = {
   184→   184→        let mut g = state.graph.lock().await;
   185→   185→        g.add(req)
   186→   186→    };
   187→   187→    // Also fold into the Holographic Context Memory (HCM)
   188→   188→    {
   189→   189→        let mut hcm = state.hcm.lock().await;
   190→   190→        let key = crate::hcm::hash_to_vector(&id, hcm.dim);
   191→   191→        let value = crate::hcm::hash_to_vector(&text, hcm.dim);
   192→   192→        hcm.fold(&key, &value);
   193→   193→    }
   194→   194→    state.stats.lock().await.hcm_pairs_folded += 1;
   195→   195→    Json(json!({ "ok": true, "id": id, "edges_created": edges_created }))
   196→   196→}

   251→   251→// ---------------------------------------------------------------------------
   252→   252→// POST /v1/interrupt — interrupt the current inference and inject new input
   253→   253→//
   254→   254→// Allows the user to interrupt the AI mid-reasoning and redirect it.
   255→   255→// The request body contains:
   256→   256→//   { "action": "interrupt", "new_instruction": "user's new directive" }
   257→   257→//
   258→   258→// When interrupted, the current pipeline's backend call is aborted (if in
   259→   259→// progress) and the new instruction is queued for the next cycle.
   260→   260→// ---------------------------------------------------------------------------
   261→   261→
   262→   262→/// Interrupt the current inference and inject a new directive.
   263→   263→///
   264→   264→/// Accepts a JSON body of the shape `{ "action": "interrupt", "new_instruction": "…" }`.
   265→   265→/// Two `action` values are supported:
   266→   266→///
   267→   267→/// - `"interrupt"` — injects `new_instruction` into the memory graph as
   268→   268→///   a high-priority `"interrupt"` node so the next retrieval picks it up.
   269→   269→///   A full implementation would also cancel the in-flight backend request
   270→   270→///   via an `AbortHandle`; the current implementation queues the directive
   271→   271→///   for the next cycle (see the inline comment in the handler body).
   272→   272→/// - `"status"` — returns whether an inference is currently in progress
   273→   273→///   (always `false` in this version; would be `true` if a pipeline is
   274→   274→///   running once the abort-handle wiring is added).
   275→   275→///
   276→   276→/// Any other `action` value yields `{ ok: false, error: "unknown action" }`.
   277→   277→pub async fn interrupt(
   278→   278→    State(state): State<AppState>,
   279→   279→    Json(body): Json<serde_json::Value>,
   280→   280→) -> impl IntoResponse {
   281→   281→    let action = body.get("action").and_then(|a| a.as_str()).unwrap_or("interrupt");
   282→   282→    let new_instruction = body.get("new_instruction").and_then(|i| i.as_str()).unwrap_or("");
   283→   283→
   284→   284→    match action {
   285→   285→        "interrupt" => {
   286→   286→            // In a full implementation, this would cancel the in-flight backend
   287→   287→            // request via an AbortHandle. For now, we set a flag that the next
   288→   288→            // pipeline check will see, and inject the new instruction as a
   289→   289→            // graph node so it's available for the next retrieval.
   290→   290→            if !new_instruction.is_empty() {
   291→   291→                let mut g = state.graph.lock().await;
   292→   292→                let id = format!("interrupt-{}", std::time::SystemTime::now()
   293→   293→                    .duration_since(std::time::UNIX_EPOCH)
   294→   294→                    .map(|d| d.as_millis())
   295→   295→                    .unwrap_or(0));
   296→   296→                g.add(crate::graph::AddNodeRequest {
   297→   297→                    id,
   298→   298→                    text: new_instruction.to_string(),
   299→   299→                    kind: "interrupt".to_string(),
   300→   300→                    metadata: serde_json::json!({ "source": "user", "priority": "high" }),
   301→   301→                });
   302→   302→            }
   303→   303→
   304→   304→            Json(json!({
   305→   305→                "ok": true,
   306→   306→                "interrupted": true,
   307→   307→                "message": "Inference interrupted. New instruction injected into the memory graph for the next cycle.",
   308→   308→                "new_instruction": new_instruction,
   309→   309→            }))
   310→   310→        }
   311→   311→        "status" => {
   312→   312→            // Check if an inference is currently in progress
   313→   313→            let stats = state.stats.lock().await;
   314→   314→            Json(json!({
   315→   315→                "ok": true,
   316→   316→                "requests": stats.requests,
   317→   317→                "active": false, // Would be true if a pipeline is running
   318→   318→            }))
   319→   319→        }
   320→   320→        _ => {
   321→   321→            Json(json!({ "ok": false, "error": "unknown action" }))
   322→   322→        }
   323→   323→    }
   324→   324→}

   326→   326→// ---------------------------------------------------------------------------
   327→   327→// POST /v1/chat/completions — the full cognitive pipeline (OpenAI-compatible)
   328→   328→// ---------------------------------------------------------------------------
   329→   329→
   330→   330→/// OpenAI-compatible chat-completions endpoint that runs the full 10-stage
   331→   331→/// cognitive pipeline.
   332→   332→///
   333→   333→/// This is the heart of Aether Engine. It accepts an OpenAI-shaped
   334→   334→/// `messages` array, augments it through the pipeline (cache → graph →
   335→   335→/// compress → complexity → decompose → solve → synthesize → ATD-verify →
   336→   336→/// distill → cache → prefetch), and returns an OpenAI-shaped
   337→   337→/// `chat.completion` JSON response.
   338→   338→///
   339→   339→/// # Complexity branching
   340→   340→///
   341→   341→/// - **Simple** → one backend call.
   342→   342→/// - **Moderate** → two backend calls: a think step, then an answer step
   343→   343→///   that consumes the think step's output.
   344→   344→/// - **Complex** → check the distillation store for a reusable
   345→   345→///   decomposition pattern; if hit, reuse it; otherwise run [`decompose`].
   346→   346→///   Solve each sub-question sequentially with dependency injection, then
   347→   347→///   synthesize.
   348→   348→///
   349→   349→/// # ATD retry
   350→   350→///
   351→   351→/// Stage 8 runs [`atd::verify`] on the synthesized response. If it fails,
   352→   352→/// a single retry is attempted with a recommendation-specific prompt
   353→   353→/// injection (`FallBackToSimpleShot` ⇒ clean simple retry; anything else
   354→   354→/// ⇒ standard retry prompt). If the retry also fails ATD, the response
   355→   355→/// with the better `collision_delta` is returned.
   356→   356→///
   357→   357→/// # Caching & prefetch
   358→   358→///
   359→   359→/// Stage 1 checks the action cache (instant return on hit). The final
   360→   360→/// response is always written back to the action cache. Stage 10 spawns a
   361→   361→/// detached [`prefetch`] task that warms the retrieval cache for the
   362→   362→/// top-3 retrieved nodes' texts so a follow-up question on the same
   363→   363→/// subgraph is fast.
   364→   364→pub async fn chat_completions(
   365→   365→    State(state): State<AppState>,
   366→   366→    Json(body): Json<serde_json::Value>,
   367→   367→) -> impl IntoResponse {
   368→   368→    let start = Instant::now();
   369→   369→    state.stats.lock().await.requests += 1;
   370→   370→
   371→   371→    // Extract the last user message as the retrieval query.
   372→   372→    let query = extract_user_query(&body);
   373→   373→
   374→   374→    // Vectorize the query against the graph's current vocabulary.
   375→   375→    let qvec: SparseVec = {
   376→   376→        let g = state.graph.lock().await;
   377→   377→        g.vectorizer.vectorize(&query)
   378→   378→    };
   379→   379→
   380→   380→    // ---- Stage 1: Action cache fast-path (similarity > 0.95) ----
   381→   381→    // The cache is checked *before* any backend work — a hit short-circuits
   382→   382→    // the entire pipeline and returns instantly.
   383→   383→    {
   384→   384→        let cache = state.cache.lock().await;
   385→   385→        if let Some((resp, _sim)) = cache.get(&query, &qvec) {
   386→   386→            // Explicitly drop the cache guard *before* re-locking stats to
   387→   387→            // avoid holding two mutexes simultaneously (which would be a
   388→   388→            // deadlock risk if the lock order were ever reversed elsewhere).
   389→   389→            drop(cache);
   390→   390→            state.stats.lock().await.cache_hits += 1;
   391→   391→            return Json(openai_completion(&resp, "aether-cache")).into_response();
   392→   392→        }
   393→   393→    }
   394→   394→
   395→   395→    // ---- Stage 2: Graph retrieval (TF-IDF + edge expansion) ----
   396→   396→    let retrieved: Vec<ScoredNode> = {
   397→   397→        let g = state.graph.lock().await;
   398→   398→        g.retrieve(&query, 8)
   399→   399→    };
   400→   400→
   401→   401→    // ---- Stage 3: Context compression (40K→4K preserving signal) ----
   402→   402→    // Two-tier: hit the retrieval cache first (warmed by the prefetcher);
   403→   403→    // on miss, run the compressor and write the result back so the next
   404→   404→    // similar query skips this expensive step.
   405→   405→    let context_block = {
   406→   406→        let mut cache = state.cache.lock().await;
   407→   407→        if let Some(ctx) = cache.get_retrieval(&query, &qvec) {
   408→   408→            ctx
   409→   409→        } else {
   410→   410→            let ctx = compress::compress(&retrieved, &query, 6000);
   411→   411→            cache.put_retrieval(&query, qvec.clone(), ctx.clone());
   412→   412→            ctx
   413→   413→        }
   414→   414→    };
   415→   415→
   416→   416→    // ---- Stage 4: Complexity analysis ----
   417→   417→    let complexity = decompose::analyze_complexity(&query);
   418→   418→    let mut pipeline = PipelineState::new(&query);
   419→   419→    pipeline.complexity = complexity.clone();
   420→   420→
   421→   421→    // ---- Stage 5-7: Cognitive decomposition + sequential solving + synthesis ----
   422→   422→    let final_response = match complexity {
   423→   423→        Complexity::Simple => {
   424→   424→            // Simple query — one-shot through the backend
   425→   425→            pipeline.stages_completed.push("one-shot".into());
   426→   426→            call_backend(&state, &body, &context_block).await
   427→   427→        }
   428→   428→        Complexity::Moderate => {
   429→   429→            // Moderate — two-step: think, then answer
   430→   430→            state.stats.lock().await.decompositions += 1;
   431→   431→            pipeline.stages_completed.push("two-step-think".into());
   432→   432→            pipeline.stages_completed.push("two-step-answer".into());
   433→   433→
   434→   434→            // Step 1: Think about the problem
   435→   435→            let think_body = inject_think_step(&body, &context_block, &query);
   436→   436→            let think_response = call_backend(&state, &think_body, &context_block).await;
   437→   437→            pipeline.total_backend_calls += 1;
   438→   438→
   439→   439→            // Step 2: Answer using the thinking result
   440→   440→            let answer_body = inject_answer_step(&body, &think_response, &context_block);
   441→   441→            call_backend(&state, &answer_body, &context_block).await
   442→   442→        }
   443→   443→        Complexity::Complex => {
   444→   444→            // Complex — full decomposition pipeline
   445→   445→            state.stats.lock().await.decompositions += 1;
   446→   446→            pipeline.stages_completed.push("decompose".into());
   447→   447→
   448→   448→            // Check distillation store for a reusable decomposition pattern
   449→   449→            let sub_questions = {
   450→   450→                let distill = state.distillation.lock().await;
   451→   451→                if let Some(subs) = distill.find(&query, &qvec) {
   452→   452→                    state.stats.lock().await.distillation_hits += 1;
   453→   453→                    pipeline.stages_completed.push("distillation-hit".into());
   454→   454→                    subs
   455→   455→                } else {
   456→   456→                    // Fresh decomposition
   457→   457→                    decompose::decompose(&query, &retrieved)
   458→   458→                }
   459→   459→            };
   460→   460→
   461→   461→            pipeline.sub_questions = sub_questions.clone();
   462→   462→            pipeline.stages_completed.push("solve".into());
   463→   463→
   464→   464→            // Solve each sub-question sequentially, injecting dependency
   465→   465→            // answers. The decomposer emits sub-questions in dependency
   466→   466→            // order (topological), so by the time we reach sub-question N
   467→   467→            // every entry it depends_on is already in the `answers` map.
   468→   468→            // The backend sees only one sub-question at a time, with a
   469→   469→            // fresh window — that's the whole point of decomposition:
   470→   470→            // the model never has to hold more than one step in its head.
   471→   471→            let mut answers: HashMap<String, String> = HashMap::new();
   472→   472→            for sub in &sub_questions {
   473→   473→                let sub_prompt = decompose::build_sub_prompt(sub, &answers);
   474→   474→                let sub_body = inject_sub_question(&body, &sub_prompt, &context_block);
   475→   475→                let sub_answer = call_backend(&state, &sub_body, &context_block).await;
   476→   476→                pipeline.total_backend_calls += 1;
   477→   477→                answers.insert(sub.id.clone(), sub_answer);
   478→   478→            }
   479→   479→
   480→   480→            // Synthesis: if there's a "synth" question, its answer is the final response
   481→   481→            pipeline.stages_completed.push("synthesize".into());
   482→   482→            if sub_questions.iter().any(|s| s.id == "synth") {
   483→   483→                answers.get("synth").cloned().unwrap_or_default()
   484→   484→            } else {
   485→   485→                // No explicit synthesis — combine all answers
   486→   486→                let combined: Vec<String> = sub_questions
   487→   487→                    .iter()
   488→   488→                    .filter_map(|s| answers.get(&s.id).cloned())
   489→   489→                    .collect();
   490→   490→                combined.join("\n\n")
   491→   491→            }
   492→   492→        }
   493→   493→    };
   494→   494→
   495→   495→    pipeline.synthesis = Some(final_response.clone());
   496→   496→
   497→   497→    // ---- Stage 8: Asymmetric Tensor Dueling (ATD) verification ----
   498→   498→    // Replaces the simple verify_response with a dual-graph collision:
   499→   499→    // Graph A (likelihood) must overcome Graph B (entropy) for validation.
   500→   500→    state.stats.lock().await.verifications += 1;
   501→   501→    state.stats.lock().await.atd_verifications += 1;
   502→   502→
   503→   503→    let atd_config = crate::atd::ATDConfig::default();
   504→   504→    let atd_result = crate::atd::verify(&final_response, &query, &atd_config);
   505→   505→
   506→   506→    let final_output = if atd_result.validated {
   507→   507→        // ATD validated — response survived the likelihood-entropy collision
   508→   508→        state.stats.lock().await.verifications_passed += 1;
   509→   509→        state.stats.lock().await.atd_validated += 1;
   510→   510→        pipeline.verification_passed = true;
   511→   511→        pipeline.stages_completed.push("atd-validated".into());
   512→   512→        final_response
   513→   513→    } else {
   514→   514→        // ATD rejected — retry based on the recommendation
   515→   515→        state.stats.lock().await.atd_rejected += 1;
   516→   516→        pipeline.stages_completed.push("atd-rejected-retry".into());
   517→   517→
   518→   518→        let retry_body = match atd_result.recommendation {
   519→   519→            crate::atd::ATDRecommendation::FallBackToSimpleShot => {
   520→   520→                // Model is confused — simplify
   521→   521→                inject_simple_retry(&body, &context_block, &query)
   522→   522→            }
   523→   523→            _ => {
   524→   524→                // Standard retry with more explicit prompt
   525→   525→                inject_retry(&body, &context_block, &query)
   526→   526→            }
   527→   527→        };
   528→   528→        let retry_response = call_backend(&state, &retry_body, &context_block).await;
   529→   529→        pipeline.total_backend_calls += 1;
   530→   530→
   531→   531→        // Re-verify the retry with ATD
   532→   532→        let retry_atd = crate::atd::verify(&retry_response, &query, &atd_config);
   533→   533→        if retry_atd.validated {
   534→   534→            state.stats.lock().await.verifications_passed += 1;
   535→   535→            state.stats.lock().await.atd_validated += 1;
   536→   536→            pipeline.verification_passed = true;
   537→   537→            pipeline.stages_completed.push("atd-retry-validated".into());
   538→   538→            retry_response
   539→   539→        } else {
   540→   540→            // Both attempts failed ATD — return the one with better collision delta
   541→   541→            state.stats.lock().await.atd_rejected += 1;
   542→   542→            pipeline.verification_passed = false;
   543→   543→            pipeline.stages_completed.push("atd-fail-final".into());
   544→   544→            if retry_atd.collision_delta > atd_result.collision_delta {
   545→   545→                retry_response
   546→   546→            } else {
   547→   547→                final_response
   548→   548→            }
   549→   549→        }
   550→   550→    };
   551→   551→
   552→   552→    pipeline.total_latency_ms = start.elapsed().as_millis() as u64;
   553→   553→
   554→   554→    // ---- Stage 9: Knowledge distillation (store successful patterns) ----
   555→   555→    // Only Complex queries get a reusable decomposition pattern. Moderate
   556→   556→    // queries use a fixed two-step template that doesn't benefit from
   557→   557→    // caching, and Simple queries don't decompose at all. The pattern is
   558→   558→    // stored only when ATD verification passed — we don't want to
   559→   559→    // distill a *failed* decomposition (it would poison future lookups).
   560→   560→    if matches!(complexity, Complexity::Complex) && pipeline.verification_passed {
   561→   561→        let mut distill = state.distillation.lock().await;
   562→   562→        distill.store(&query, qvec.clone(), pipeline.sub_questions.clone());
   563→   563→    }
   564→   564→
   565→   565→    // ---- Cache the final response ----
   566→   566→    state.cache.lock().await.put(&query, qvec.clone(), final_output.clone());
   567→   567→
   568→   568→    // ---- Stage 10: Speculative prefetch ----
   569→   569→    prefetch(state.clone(), &retrieved).await;
   570→   570→
   571→   571→    Json(openai_completion(&final_output, "aether-pipeline")).into_response()
   572→   572→}

   578→   578→/// Forward a request to the GGUF backend, returning the assistant's text response.
   579→   579→/// Falls back to a graph-only response if the backend is unreachable.
   580→   580→async fn call_backend(
   581→   581→    state: &AppState,
   582→   582→    body: &serde_json::Value,
   583→   583→    context_block: &str,
   584→   584→) -> String {
   585→   585→    let augmented = augment_messages(body, context_block);
   586→   586→    let backend_url = format!("{}/chat/completions", state.backend.trim_end_matches('/'));
   587→   587→    let client = state.client.clone();
   588→   588→
   589→   589→    match client.post(&backend_url).json(&augmented).send().await {
   590→   590→        Ok(resp) if resp.status().is_success() => {
   591→   591→            let txt = resp.text().await.unwrap_or_else(|_| "{}".to_string());
   592→   592→            extract_assistant_content(&txt).unwrap_or_else(|| {
   593→   593→                fallback_response(context_block, "empty backend response")
   594→   594→            })
   595→   595→        }
   596→   596→        Ok(resp) => {
   597→   597→            let status = resp.status();
   598→   598→            let body_txt = resp.text().await.unwrap_or_default();
   599→   599→            fallback_response(
   600→   600→                context_block,
   601→   601→                &format!("backend returned {status}: {}", body_txt.chars().take(200).collect::<String>()),
   602→   602→            )
   603→   603→        }
   604→   604→        Err(e) => {
   605→   605→            fallback_response(context_block, &format!("backend unreachable: {e}"))
   606→   606→        }
   607→   607→    }
   608→   608→}

   614→   614→/// Inject the Aether context block into the system message.
   615→   615→fn augment_messages(body: &serde_json::Value, context_block: &str) -> serde_json::Value {
   616→   616→    let mut out = body.clone();
   617→   617→    let prompt = aether_prompt(context_block);
   618→   618→    if let Some(messages) = out.get_mut("messages").and_then(|m| m.as_array_mut()) {
   619→   619→        let mut injected = false;
   620→   620→        for m in messages.iter_mut() {
   621→   621→            if m.get("role").and_then(|r| r.as_str()) == Some("system") && !injected {
   622→   622→                let existing = m
   623→   623→                    .get("content")
   624→   624→                    .and_then(|c| c.as_str())
   625→   625→                    .unwrap_or("")
   626→   626→                    .to_string();
   627→   627→                let new_content = format!("{}\n\n{}", prompt, existing);
   628→   628→                if let Some(obj) = m.as_object_mut() {
   629→   629→                    obj.insert("content".to_string(), serde_json::Value::String(new_content));
   630→   630→                }
   631→   631→                injected = true;
   632→   632→            }
   633→   633→        }
   634→   634→        if !injected {
   635→   635→            messages.insert(0, json!({ "role": "system", "content": prompt }));
   636→   636→        }
   637→   637→    }
   638→   638→    out
   639→   639→}

   672→   672→/// Inject a sub-question as the primary user message for decomposition.
   673→   673→fn inject_sub_question(body: &serde_json::Value, sub_prompt: &str, context_block: &str) -> serde_json::Value {
   674→   674→    let mut out = body.clone();
   675→   675→    let sub_system = format!(
   676→   676→        "{context_block}\n\n\
   677→   677→         # AETHER COGNITIVE PIPELINE — SUB-QUESTION\n\
   678→   678→         You are solving one step of a larger problem. Focus ONLY on this sub-question.\n\
   679→   679→         Provide a clear, concise answer."
   680→   680→    );
   681→   681→    if let Some(messages) = out.get_mut("messages").and_then(|m| m.as_array_mut()) {
   682→   682→        // Replace the system message
   683→   683→        if let Some(sys) = messages.get_mut(0) {
   684→   684→            if sys.get("role").and_then(|r| r.as_str()) == Some("system") {
   685→   685→                if let Some(obj) = sys.as_object_mut() {
   686→   686→                    obj.insert("content".to_string(), serde_json::Value::String(sub_system));
   687→   687→                }
   688→   688→            }
   689→   689→        }
   690→   690→        // Replace the last user message with the sub-question
   691→   691→        if let Some(last) = messages.last_mut() {
   692→   692→            if last.get("role").and_then(|r| r.as_str()) == Some("user") {
   693→   693→                if let Some(obj) = last.as_object_mut() {
   694→   694→                    obj.insert("content".to_string(), serde_json::Value::String(sub_prompt.to_string()));
   695→   695→                }
   696→   696→            }
   697→   697→        }
   698→   698→    }
   699→   699→    out
   700→   700→}

   755→   755→/// Extract the last user-message content from an OpenAI-shaped `messages`
   756→   756→/// array. Used as the retrieval query for Stage 2 (graph retrieval) and
   757→   757→/// as the cache key for Stage 1 (action cache). Walks the messages in
   758→   758→/// reverse so the *most recent* user turn wins (handles multi-turn chats
   759→   759→/// where the user has refined their question across several messages).
   760→   760→fn extract_user_query(body: &serde_json::Value) -> String {
   761→   761→    let messages = match body.get("messages").and_then(|m| m.as_array()) {
   762→   762→        Some(m) => m,
   763→   763→        None => return String::new(),
   764→   764→    };
   765→   765→    for m in messages.iter().rev() {
   766→   766→        if m.get("role").and_then(|r| r.as_str()) == Some("user") {
   767→   767→            if let Some(c) = m.get("content") {
   768→   768→                return content_to_string(c);
   769→   769→            }
   770→   770→        }
   771→   771→    }
   772→   772→    String::new()
   773→   773→}
   774→   774→
   775→   775→/// Normalize a `content` field (which the OpenAI spec allows to be either
   776→   776→/// a plain string or an array of `{type, text}` parts) into a single
   777→   777→/// space-joined string. Non-text parts (images, tool calls, …) are
   778→   778→/// dropped because the retrieval pipeline is text-only.
   779→   779→fn content_to_string(c: &serde_json::Value) -> String {
   780→   780→    match c {
   781→   781→        serde_json::Value::String(s) => s.clone(),
   782→   782→        serde_json::Value::Array(arr) => arr
   783→   783→            .iter()
   784→   784→            .filter_map(|p| {
   785→   785→                if p.get("type").and_then(|t| t.as_str()) == Some("text") {
   786→   786→                    p.get("text").and_then(|t| t.as_str()).map(|s| s.to_string())
   787→   787→                } else {
   788→   788→                    None
   789→   789→                }
   790→   790→            })
   791→   791→            .collect::<Vec<_>>()
   792→   792→            .join(" "),
   793→   793→        _ => String::new(),
   794→   794→    }
   795→   795→}

   797→   797→/// Extract the assistant's text content from an OpenAI-shaped chat
   798→   798→/// completion response body. Returns `None` if the body isn't valid JSON
   799→   799→/// or doesn't contain the expected `choices[0].message.content` path —
   800→   800→/// the caller ([`call_backend`]) turns that into a fallback response.
   801→   801→fn extract_assistant_content(body: &str) -> Option<String> {
   802→   802→    let v: serde_json::Value = serde_json::from_str(body).ok()?;
   803→   803→    v.get("choices")
   804→   804→        .and_then(|c| c.get(0))
   805→   805→        .and_then(|c| c.get("message"))
   806→   806→        .and_then(|m| m.get("content"))
   807→   807→        .and_then(|c| c.as_str())
   808→   808→        .map(|s| s.to_string())
   809→   809→}

   811→   811→/// Build an OpenAI-shaped `chat.completion` JSON response from a plain
   812→   812→/// text content string. Used by every code path that returns to the

   814→   814→/// set to a synthetic name (`"aether-cache"`, `"aether-pipeline"`, …)

   821→   821→        "object": "chat.completion",
   822→   822→        "created": chrono_now(),
   823→   823→        "model": model,
   824→   824→        "choices": [{
   825→   825→            "index": 0,
   826→   826→            "message": { "role": "assistant", "content": content },
   827→   827→            "finish_reason": "stop"
   828→   828→        }],
   829→   829→        "usage": { "prompt_tokens": 0, "completion_tokens": 0, "total_tokens": 0 }
   830→   830→    })
   831→   831→}

   874→   874→/// Speculative prefetch: warm the retrieval cache for graph-adjacent candidate
   875→   875→/// queries so the next, related query is instant.
   876→   876→async fn prefetch(state: AppState, retrieved: &[ScoredNode]) {
   877→   877→    if retrieved.is_empty() {
   878→   878→        return;
   879→   879→    }
   880→   880→    let graph = state.graph.clone();
   881→   881→    let cache = state.cache.clone();
   882→   882→    let candidates: Vec<String> = retrieved.iter().take(3).map(|n| n.text.clone()).collect();
   883→   883→    tokio::spawn(async move {
   884→   884→        for q in candidates {
   885→   885→            let (qvec, ctx) = {
   886→   886→                let g = graph.lock().await;
   887→   887→                let qv = g.vectorizer.vectorize(&q);
   888→   888→                let r = g.retrieve(&q, 8);
   889→   889→                let c = compress::compress(&r, &q, 6000);
   890→   890→                (qv, c)
   891→   891→            };
   892→   892→            cache.lock().await.put_retrieval(&q, qvec, ctx);
   893→   893→        }
   894→   894→    });
   895→   895→}

   906→   906→// POST /v1/chat/stream — Server-Sent Events chat stream
   907→   907→// ---------------------------------------------------------------------------
   908→   908→
   909→   909→/// OpenAI-compatible chat-completions **streaming** endpoint.
   910→   910→///
   911→   911→/// Accepts the same body shape as [`chat_completions`] (`{ messages: [...] }`)
   912→   912→/// and returns a `text/event-stream` response. Each token of the final
   913→   913→/// assistant message is emitted as an SSE event of the form:
   914→   914→///
   915→   915→/// ```text
   916→   916→/// event: token
   917→   917→/// data: {"token": "hello"}
   918→   918→///
   919→   919→/// ```
   920→   920→///
   921→   921→/// A terminal `event: done` event signals end-of-stream.
   922→   922→///
   923→   923→/// # Implementation note
   924→   924→///
   925→   925→/// Aether Engine's backend HTTP client is not natively streaming (it would
   926→   926→/// require a chunked-transfer-aware `reqwest::Body` adapter), so this handler
   927→   927→/// runs the simplified pipeline (graph retrieval → compression → one
   928→   928→/// `call_backend` round-trip) and then **emits the final response token by
   929→   929→/// token** as SSE events. The browser's `EventSource` parser sees one event
   930→   930→/// per token; the only difference from a true token-stream is that all events
   931→   931→/// arrive in a single HTTP body rather than being interleaved with backend
   932→   932→/// generation. This is the fallback shape explicitly allowed by the spec
   933→   933→/// ("simple chunked HTTP response with `text/event-stream` content type").
   934→   934→pub async fn chat_completions_stream(
   935→   935→    State(state): State<AppState>,
   936→   936→    Json(body): Json<serde_json::Value>,
   937→   937→) -> impl IntoResponse {
   938→   938→    state.stats.lock().await.requests += 1;
   939→   939→    let query = extract_user_query(&body);
   940→   940→
   941→   941→    // Stage 2: graph retrieval (TF-IDF + edge expansion).
   942→   942→    let retrieved: Vec<ScoredNode> = {
   943→   943→        let g = state.graph.lock().await;
   944→   944→        g.retrieve(&query, 8)
   945→   945→    };
   946→   946→    // Stage 3: context compression.
   947→   947→    let context_block = compress::compress(&retrieved, &query, 6000);
   948→   948→
   949→   949→    // Simplified pipeline: one backend call. The full chat_completions
   950→   950→    // pipeline (decomposition / ATD / distillation) is intentionally not
   951→   951→    // duplicated here — the streaming endpoint is for live UX, the regular
   952→   952→    // endpoint is for highest-quality responses.
   953→   953→    let response = call_backend(&state, &body, &context_block).await;
   954→   954→
   955→   955→    // Build the SSE body: one `event: token` per whitespace-split token of
   956→   956→    // the final response, terminated by `event: done`.
   957→   957→    let mut sse = String::with_capacity(response.len().saturating_mul(2));
   958→   958→    for token in response.split_whitespace() {
   959→   959→        // JSON-encode the token (handles embedded quotes / backslashes) and
   960→   960→        // interpolate it into the `data:` line. The leading space after
   961→   961→        // `data:` is required by the SSE spec.
   962→   962→        let token_json = serde_json::to_string(token).unwrap_or_else(|_| "\"\"".to_string());
   963→   963→        sse.push_str(&format!(
   964→   964→            "event: token\ndata: {{\"token\": {}}}\n\n",
   965→   965→            token_json
   966→   966→        ));
   967→   967→    }
   968→   968→    sse.push_str("event: done\ndata: {\"done\": true}\n\n");
   969→   969→
   970→   970→    (
   971→   971→        axum::http::StatusCode::OK,
   972→   972→        [("content-type", "text/event-stream")],
   973→   973→        sse,
   974→   974→    )
   975→   975→}

  1241→  1241→    let edge_count = g.edge_count();

  1243→  1243→        .adjacency
  1244→  1244→        .values()

  1247→  1247→        .unwrap_or(0);
  1248→  1248→    let avg_edges_per_node = if node_count > 0 {
  1249→  1249→        edge_count as f64 / node_count as f64

  1251→  1251→        0.0
  1252→  1252→    };
  1253→  1253→    let density = if node_count > 1 {
  1254→  1254→        edge_count as f64 / (node_count as f64 * (node_count as f64 - 1.0))
  1255→  1255→    } else {
  1256→  1256→        0.0
  1257→  1257→    };
  1258→  1258→
  1259→  1259→    // Top connected nodes: sort adjacency entries by length (desc), take 5.
  1260→  1260→    // We clone the (id, edge_count) pairs out of the adjacency map so we can
  1261→  1261→    // drop the borrow before looking up node texts (avoiding a double-borrow
  1262→  1262→    // of `g.adjacency` and `g.nodes`).
  1263→  1263→    let mut node_edge_counts: Vec<(String, usize)> = g
  1264→  1264→        .adjacency
  1265→  1265→        .iter()
  1266→  1266→        .map(|(id, nbrs)| (id.clone(), nbrs.len()))
  1267→  1267→        .collect();
  1268→  1268→    node_edge_counts.sort_by(|a, b| b.1.cmp(&a.1));
  1269→  1269→
  1270→  1270→    let top_connected: Vec<serde_json::Value> = node_edge_counts
  1271→  1271→        .into_iter()
  1272→  1272→        .take(5)
  1273→  1273→        .filter_map(|(id, ec)| {
  1274→  1274→            g.nodes.get(&id).map(|n| {
  1275→  1275→                json!({
  1276→  1276→                    "id": id,
  1277→  1277→                    "text": n.text.clone(),
  1278→  1278→                    "edge_count": ec,
  1279→  1279→                })
  1280→  1280→            })
  1281→  1281→        })
  1282→  1282→        .collect();
  1283→  1283→
  1284→  1284→    Json(json!({
  1285→  1285→        "node_count": node_count,
  1286→  1286→        "edge_count": edge_count,
  1287→  1287→        "avg_edges_per_node": avg_edges_per_node,
  1288→  1288→        "max_edges": max_edges,
  1289→  1289→        "density": density,
  1290→  1290→        "top_connected_nodes": top_connected,
  1291→  1291→    }))
  1292→  1292→}

  1303→  1303→// POST /v1/agent/run — run the autonomous agent loop
  1304→  1304→// ---------------------------------------------------------------------------
  1305→  1305→
  1306→  1306→/// Run the autonomous agent loop for a single goal.
  1307→  1307→///
  1308→  1308→/// Accepts a JSON body of the shape:
  1309→  1309→///
  1310→  1310→/// ```json
  1311→  1311→/// {
  1312→  1312→///   "goal": "set up a Python dev environment and write a hello-world script",
  1313→  1313→///   "context": { "windows": [...], "memory": [...], "plan": {...} },
  1314→  1314→///   "max_iterations": 20
  1315→  1315→/// }
  1316→  1316→/// ```
  1317→  1317→///
  1318→  1318→/// - `goal` (required) — natural-language goal the agent should achieve.
  1319→  1319→/// - `context` (optional, default `{}`) — current OS state, pretty-printed
  1320→  1320→///   into the system prompt so the LLM perceives open windows, memory,
  1321→  1321→///   active plan, etc.
  1322→  1322→/// - `max_iterations` (optional, default `20`, clamped to `[1, 50]`) —
  1323→  1323→///   hard cap on think iterations.
  1324→  1324→///
  1325→  1325→/// Returns:
  1326→  1326→///
  1327→  1327→/// ```json
  1328→  1328→/// {
  1329→  1329→///   "ok": true,
  1330→  1330→///   "result": "<final assistant text>",
  1331→  1331→///   "iterations": 3,
  1332→  1332→///   "completed": true,
  1333→  1333→///   "tool_calls": [
  1334→  1334→///     { "name": "file_read", "params": { "path": "/etc/hostname" } },
  1335→  1335→///     ...
  1336→  1336→///   ]
  1337→  1337→/// }
  1338→  1338→/// ```
  1339→  1339→///
  1340→  1340→/// The `tool_calls` array is the full transcript of every tool call the
  1341→  1341→/// LLM made across all iterations, in execution order. The Next.js layer
  1342→  1342→/// uses this transcript to perform the real OS side-effects (file IO,
  1343→  1343→/// PTY exec, window manager mutations, …) — the Aether Engine's tool
  1344→  1344→/// executors return placeholders by design (see [`crate::tools`]).
  1345→  1345→///
  1346→  1346→/// On a missing `goal` field the handler returns
  1347→  1347→/// `{ ok: false, error: "missing 'goal' field" }` without starting the
  1348→  1348→/// loop.
  1349→  1349→pub async fn agent_run(
  1350→  1350→    State(state): State<AppState>,
  1351→  1351→    Json(body): Json<serde_json::Value>,
  1352→  1352→) -> impl IntoResponse {
  1353→  1353→    let goal = body
  1354→  1354→        .get("goal")
  1355→  1355→        .and_then(|g| g.as_str())
  1356→  1356→        .unwrap_or("")
  1357→  1357→        .to_string();
  1358→  1358→    let context = body.get("context").cloned().unwrap_or(json!({}));
  1359→  1359→    let max_iterations = body
  1360→  1360→        .get("max_iterations")
  1361→  1361→        .and_then(|m| m.as_u64())
  1362→  1362→        .map(|n| n as usize)
  1363→  1363→        .unwrap_or(crate::agent::DEFAULT_MAX_ITERATIONS);
  1364→  1364→
  1365→  1365→    if goal.is_empty() {
  1366→  1366→        return Json(json!({
  1367→  1367→            "ok": false,
  1368→  1368→            "error": "missing 'goal' field",
  1369→  1369→        }))
  1370→  1370→        .into_response();
  1371→  1371→    }
  1372→  1372→
  1373→  1373→    let result = crate::agent::run_agent_loop(state, &goal, &context, max_iterations).await;
  1374→  1374→
  1375→  1375→    Json(json!({
  1376→  1376→        "ok": true,
  1377→  1377→        "result": result.result,
  1378→  1378→        "iterations": result.iterations,
  1379→  1379→        "completed": result.completed,
  1380→  1380→        "tool_calls": result.tool_calls.iter().map(|c| json!({
  1381→  1381→            "name": c.name,
  1382→  1382→            "params": c.params,
  1383→  1383→        })).collect::<Vec<_>>(),
  1384→  1384→    }))
  1385→  1385→    .into_response()
  1386→  1386→}
  1387→  1387→

  1389→  1389→// GET /v1/tools — list available agent tools (introspection)

  1393→  1393→/// description, and JSON-Schema parameters.
  1394→  1394→///
  1395→  1395→/// Used by the Next.js layer (and by `curl`) to introspect the tool
  1396→  1396→/// surface without reading the Rust source. Returns:
  1397→  1397→///
  1398→  1398→/// ```json
  1399→  1399→/// {

  1401→  1401→///   "tools": [

  1404→  1404→///       "description": "Read the full contents of a file from disk.",
  1405→  1405→///       "parameters": {
  1406→  1406→///         "type": "object",
  1407→  1407→///         "properties": { "path": { "type": "string", "..." } },
  1408→  1408→///         "required": ["path"]
  1409→  1409→///       }
  1410→  1410→///     },
  1411→  1411→///     ...
  1412→  1412→///   ]

  1414→  1414→/// ```

  1420→  1420→    let registry = crate::tools::ToolRegistry::new();
  1421→  1421→    let catalog = registry.catalog();

  1426→  1426→            "description": t.description,
  1427→  1427→            "parameters": t.parameters,
  1428→  1428→        })).collect::<Vec<_>>(),
  1429→  1429→    }))
  1430→  1430→}
