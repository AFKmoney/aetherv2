   360Ôćĺ   360Ôćĺ/// Parse zero or more [`ToolCall`]s from the LLM's response text.
   361Ôćĺ   361Ôćĺ///
   362Ôćĺ   362Ôćĺ/// Supports every common emission format small GGUF models produce:
   363Ôćĺ   363Ôćĺ///
   364Ôćĺ   364Ôćĺ/// - **Inline JSON** ÔÇö `{"tool": "file_read", "params": {"path": "ÔÇŽ"}}`
   365Ôćĺ   365Ôćĺ///   appearing anywhere in the response (including inside prose).
   366Ôćĺ   366Ôćĺ/// - **Fenced code blocks** ÔÇö ` ```json\n{ÔÇŽ}\n``` ` or ` ```tool\n{ÔÇŽ}\n``` `.
   367Ôćĺ   367Ôćĺ///   The fenced content is scanned with the same JSON extractor.
   368Ôćĺ   368Ôćĺ/// - **OpenAI function-call shape** ÔÇö `{"function": {"name": "ÔÇŽ",
   369Ôćĺ   369Ôćĺ///   "arguments": "ÔÇŽ"}}` (arguments may be a JSON string or object).
   370Ôćĺ   370Ôćĺ/// - **Wrapper arrays** ÔÇö `{"tool_calls": [{ÔÇŽ}, {ÔÇŽ}]}` expands to
   371Ôćĺ   371Ôćĺ///   multiple calls.
   372Ôćĺ   372Ôćĺ/// - **Alternate field names** ÔÇö `name`/`tool_name` for the tool name;
   373Ôćĺ   373Ôćĺ///   `arguments`/`args` for parameters.
   374Ôćĺ   374Ôćĺ///
   375Ôćĺ   375Ôćĺ/// The parser walks the response once, scanning for `{` and attempting to
   376Ôćĺ   376Ôćĺ/// extract a balanced JSON object at each occurrence. String-literal
   377Ôćĺ   377Ôćĺ/// braces (e.g. inside a `"```json\nÔÇŽ\n```"` code fence description) are
   378Ôćĺ   378Ôćĺ/// correctly skipped via the in-string / escape flags. Every successfully
   379Ôćĺ   379Ôćĺ/// parsed JSON object is then classified by [`value_to_tool_calls`] as a
   380Ôćĺ   380Ôćĺ/// single call, an array of calls, or a non-tool object (ignored).
   381Ôćĺ   381Ôćĺ///
   382Ôćĺ   382Ôćĺ/// Returns the calls in source order. Duplicates are preserved (the LLM
   383Ôćĺ   383Ôćĺ/// may legitimately emit the same call twice if the first attempt didn't
   384Ôćĺ   384Ôćĺ/// return the expected result).
   385Ôćĺ   385Ôćĺpub fn parse_tool_calls(response: &str) -> Vec<ToolCall> {
   386Ôćĺ   386Ôćĺ    let mut calls = Vec::new();
   387Ôćĺ   387Ôćĺ    let bytes = response.as_bytes();
   388Ôćĺ   388Ôćĺ    let mut i = 0;
   389Ôćĺ   389Ôćĺ    while i < bytes.len() {
   390Ôćĺ   390Ôćĺ        if bytes[i] == b'{' {
   391Ôćĺ   391Ôćĺ            if let Some((json_str, end)) = extract_balanced_json(response, i) {
   392Ôćĺ   392Ôćĺ                if let Ok(v) = serde_json::from_str::<Value>(&json_str) {
   393Ôćĺ   393Ôćĺ                    calls.extend(value_to_tool_calls(&v));
   394Ôćĺ   394Ôćĺ                }
   395Ôćĺ   395Ôćĺ                // Skip past the balanced object whether or not it parsed.
   396Ôćĺ   396Ôćĺ                // (A failed parse means malformed JSON; we still want to
   397Ôćĺ   397Ôćĺ                // jump past it to avoid re-scanning its interior.)
   398Ôćĺ   398Ôćĺ                i = end;
   399Ôćĺ   399Ôćĺ                continue;
   400Ôćĺ   400Ôćĺ            }
   401Ôćĺ   401Ôćĺ        }
   402Ôćĺ   402Ôćĺ        i += 1;
   403Ôćĺ   403Ôćĺ    }
   404Ôćĺ   404Ôćĺ    calls
   405Ôćĺ   405Ôćĺ}

   407Ôćĺ   407Ôćĺ/// Extract a balanced JSON object (including any nested braces) starting
   408Ôćĺ   408Ôćĺ/// at `start` (which MUST point at a `{` byte). Returns the matched
   409Ôćĺ   409Ôćĺ/// substring and the index *after* the closing `}`, or `None` if no
   410Ôćĺ   410Ôćĺ/// balanced match is found before end-of-input.
   411Ôćĺ   411Ôćĺ///
   412Ôćĺ   412Ôćĺ/// Honors JSON string literals (`"ÔÇŽ"`) and escape sequences (`\"`, `\\`,
   413Ôćĺ   413Ôćĺ/// ÔÇŽ) so braces inside strings don't affect depth counting.
   414Ôćĺ   414Ôćĺfn extract_balanced_json(s: &str, start: usize) -> Option<(String, usize)> {
   415Ôćĺ   415Ôćĺ    let bytes = s.as_bytes();
   416Ôćĺ   416Ôćĺ    if start >= bytes.len() || bytes[start] != b'{' {
   417Ôćĺ   417Ôćĺ        return None;
   418Ôćĺ   418Ôćĺ    }
   419Ôćĺ   419Ôćĺ    let mut depth: i32 = 0;
   420Ôćĺ   420Ôćĺ    let mut in_string = false;
   421Ôćĺ   421Ôćĺ    let mut escape = false;
   422Ôćĺ   422Ôćĺ    for i in start..bytes.len() {
   423Ôćĺ   423Ôćĺ        let c = bytes[i];
   424Ôćĺ   424Ôćĺ        if in_string {
   425Ôćĺ   425Ôćĺ            if escape {
   426Ôćĺ   426Ôćĺ                escape = false;
   427Ôćĺ   427Ôćĺ            } else if c == b'\\' {
   428Ôćĺ   428Ôćĺ                escape = true;
   429Ôćĺ   429Ôćĺ            } else if c == b'"' {
   430Ôćĺ   430Ôćĺ                in_string = false;
   431Ôćĺ   431Ôćĺ            }
   432Ôćĺ   432Ôćĺ        } else {
   433Ôćĺ   433Ôćĺ            match c {
   434Ôćĺ   434Ôćĺ                b'"' => in_string = true,
   435Ôćĺ   435Ôćĺ                b'{' => depth += 1,
   436Ôćĺ   436Ôćĺ                b'}' => {
   437Ôćĺ   437Ôćĺ                    depth -= 1;
   438Ôćĺ   438Ôćĺ                    if depth == 0 {
   439Ôćĺ   439Ôćĺ                        return Some((s[start..=i].to_string(), i + 1));
   440Ôćĺ   440Ôćĺ                    }
   441Ôćĺ   441Ôćĺ                }
   442Ôćĺ   442Ôćĺ                _ => {}
   443Ôćĺ   443Ôćĺ            }
   444Ôćĺ   444Ôćĺ        }
   445Ôćĺ   445Ôćĺ    }
   446Ôćĺ   446Ôćĺ    None
   447Ôćĺ   447Ôćĺ}
   448Ôćĺ   448Ôćĺ
   449Ôćĺ   449Ôćĺ/// Classify a parsed JSON value as zero or more tool calls. Handles single
   450Ôćĺ   450Ôćĺ/// calls, call arrays, and the `{"tool_calls": [...]}` wrapper.
   451Ôćĺ   451Ôćĺfn value_to_tool_calls(v: &Value) -> Vec<ToolCall> {
   452Ôćĺ   452Ôćĺ    // Array of calls: [{...}, {...}]
   453Ôćĺ   453Ôćĺ    if let Some(arr) = v.as_array() {
   454Ôćĺ   454Ôćĺ        return arr.iter().filter_map(value_to_single_call).collect();
   455Ôćĺ   455Ôćĺ    }
   456Ôćĺ   456Ôćĺ    // Wrapper: {"tool_calls": [...]} or {"calls": [...]}
   457Ôćĺ   457Ôćĺ    for wrapper_key in ["tool_calls", "calls"] {
   458Ôćĺ   458Ôćĺ        if let Some(inner) = v.get(wrapper_key).and_then(|x| x.as_array()) {
   459Ôćĺ   459Ôćĺ            return inner.iter().filter_map(value_to_single_call).collect();
   460Ôćĺ   460Ôćĺ        }
   461Ôćĺ   461Ôćĺ    }
   462Ôćĺ   462Ôćĺ    // Single call
   463Ôćĺ   463Ôćĺ    if let Some(c) = value_to_single_call(v) {
   464Ôćĺ   464Ôćĺ        return vec![c];
   465Ôćĺ   465Ôćĺ    }
   466Ôćĺ   466Ôćĺ    Vec::new()
   467Ôćĺ   467Ôćĺ}
   468Ôćĺ   468Ôćĺ
   469Ôćĺ   469Ôćĺ/// Convert a single JSON object to a [`ToolCall`]. Accepts the field-name
   470Ôćĺ   470Ôćĺ/// aliases documented on [`parse_tool_calls`]. Returns `None` when the
   471Ôćĺ   471Ôćĺ/// object doesn't look like a tool call (no `tool`/`name`/`tool_name`/
   472Ôćĺ   472Ôćĺ/// `function.name` field).
   473Ôćĺ   473Ôćĺfn value_to_single_call(v: &Value) -> Option<ToolCall> {
   474Ôćĺ   474Ôćĺ    // Direct format: {"tool": "...", "params": {...}}
   475Ôćĺ   475Ôćĺ    let name = v
   476Ôćĺ   476Ôćĺ        .get("tool")
   477Ôćĺ   477Ôćĺ        .and_then(|x| x.as_str())
   478Ôćĺ   478Ôćĺ        .or_else(|| v.get("name").and_then(|x| x.as_str()))
   479Ôćĺ   479Ôćĺ        .or_else(|| v.get("tool_name").and_then(|x| x.as_str()));
   480Ôćĺ   480Ôćĺ
   481Ôćĺ   481Ôćĺ    if let Some(name) = name {
   482Ôćĺ   482Ôćĺ        let params = v
   483Ôćĺ   483Ôćĺ            .get("params")
   484Ôćĺ   484Ôćĺ            .cloned()
   485Ôćĺ   485Ôćĺ            .or_else(|| v.get("arguments").cloned())
   486Ôćĺ   486Ôćĺ            .or_else(|| v.get("args").cloned())
   487Ôćĺ   487Ôćĺ            .filter(Value::is_object)
   488Ôćĺ   488Ôćĺ            .unwrap_or_else(|| json!({}));
   489Ôćĺ   489Ôćĺ        return Some(ToolCall {
   490Ôćĺ   490Ôćĺ            name: name.to_string(),
   491Ôćĺ   491Ôćĺ            params,
   492Ôćĺ   492Ôćĺ        });
   493Ôćĺ   493Ôćĺ    }
   494Ôćĺ   494Ôćĺ
   495Ôćĺ   495Ôćĺ    // OpenAI function-call format: {"id": "...", "function": {"name": "...", "arguments": "..."}}
   496Ôćĺ   496Ôćĺ    if let Some(func) = v.get("function") {
   497Ôćĺ   497Ôćĺ        if let Some(fname) = func.get("name").and_then(|x| x.as_str()) {
   498Ôćĺ   498Ôćĺ            let params = match func.get("arguments") {
   499Ôćĺ   499Ôćĺ                // OpenAI ships arguments as a JSON *string* ÔÇö parse it.
   500Ôćĺ   500Ôćĺ                Some(Value::String(s)) => {
   501Ôćĺ   501Ôćĺ                    serde_json::from_str(s).unwrap_or_else(|_| json!({}))
   502Ôćĺ   502Ôćĺ                }
   503Ôćĺ   503Ôćĺ                Some(other) if other.is_object() => other.clone(),
   504Ôćĺ   504Ôćĺ                _ => json!({}),
   505Ôćĺ   505Ôćĺ            };
   506Ôćĺ   506Ôćĺ            return Some(ToolCall {
   507Ôćĺ   507Ôćĺ                name: fname.to_string(),
   508Ôćĺ   508Ôćĺ                params,
   509Ôćĺ   509Ôćĺ            });
   510Ôćĺ   510Ôćĺ        }
   511Ôćĺ   511Ôćĺ    }
   512Ôćĺ   512Ôćĺ
   513Ôćĺ   513Ôćĺ    None
   514Ôćĺ   514Ôćĺ}

   576Ôćĺ   576Ôćĺ// The main agent loop
   577Ôćĺ   577Ôćĺ// ---------------------------------------------------------------------------
   578Ôćĺ   578Ôćĺ
   579Ôćĺ   579Ôćĺ/// Run the autonomous agent loop until termination.
   580Ôćĺ   580Ôćĺ///
   581Ôćĺ   581Ôćĺ/// # Arguments
   582Ôćĺ   582Ôćĺ///
   583Ôćĺ   583Ôćĺ/// - `state` ÔÇö the engine's shared [`AppState`] (used for the backend LLM
   584Ôćĺ   584Ôćĺ///   call only; no other engine state is touched).
   585Ôćĺ   585Ôćĺ/// - `goal` ÔÇö the user's natural-language goal (e.g. *"set up a Python
   586Ôćĺ   586Ôćĺ///   dev environment and write a hello-world script"*).
   587Ôćĺ   587Ôćĺ/// - `context` ÔÇö the current OS state as a JSON object (open windows,
   588Ôćĺ   588Ôćĺ///   memory snapshot, active plan, goals, ÔÇŽ). Pretty-printed into the
   589Ôćĺ   589Ôćĺ///   system prompt so the LLM perceives it.
   590Ôćĺ   590Ôćĺ/// - `max_iterations` ÔÇö iteration cap. Clamped to `[1, ABSOLUTE_MAX_ITERATIONS]`.
   591Ôćĺ   591Ôćĺ///
   592Ôćĺ   592Ôćĺ/// # Returns
   593Ôćĺ   593Ôćĺ///
   594Ôćĺ   594Ôćĺ/// An [`AgentRunResult`] carrying the final assistant text, the iteration
   595Ôćĺ   595Ôćĺ/// count, the full transcript of tool calls, and the `completed` flag.
   596Ôćĺ   596Ôćĺ///
   597Ôćĺ   597Ôćĺ/// # Termination
   598Ôćĺ   598Ôćĺ///
   599Ôćĺ   599Ôćĺ/// See the module docs. The loop stops on `TASK_COMPLETE`, on a
   600Ôćĺ   600Ôćĺ/// no-tool-calls response, or on hitting `max_iterations`.
   601Ôćĺ   601Ôćĺ///
   602Ôćĺ   602Ôćĺ/// # Errors
   603Ôćĺ   603Ôćĺ///
   604Ôćĺ   604Ôćĺ/// This function never returns `Result::Err` ÔÇö every failure path
   605Ôćĺ   605Ôćĺ/// (backend unreachable, malformed response, unknown tool, invalid
   606Ôćĺ   606Ôćĺ/// params, executor error) is converted into a diagnostic string and fed
   607Ôćĺ   607Ôćĺ/// back into the message stream or surfaced in the final result. The
   608Ôćĺ   608Ôćĺ/// caller decides how to react.
   609Ôćĺ   609Ôćĺpub async fn run_agent_loop(
   610Ôćĺ   610Ôćĺ    state: AppState,
   611Ôćĺ   611Ôćĺ    goal: &str,
   612Ôćĺ   612Ôćĺ    context: &Value,
   613Ôćĺ   613Ôćĺ    max_iterations: usize,
   614Ôćĺ   614Ôćĺ) -> AgentRunResult {
   615Ôćĺ   615Ôćĺ    // Clamp the iteration cap. We accept the caller's value but enforce a
   616Ôćĺ   616Ôćĺ    // hard ceiling so a buggy caller can't pin a CPU forever against a
   617Ôćĺ   617Ôćĺ    // broken backend.
   618Ôćĺ   618Ôćĺ    let cap = max_iterations.clamp(1, ABSOLUTE_MAX_ITERATIONS);
   619Ôćĺ   619Ôćĺ
   620Ôćĺ   620Ôćĺ    // Build the tool registry + system prompt once; both are reused on
   621Ôćĺ   621Ôćĺ    // every iteration.
   622Ôćĺ   622Ôćĺ    let registry = ToolRegistry::new();
   623Ôćĺ   623Ôćĺ    let catalog = registry.catalog();
   624Ôćĺ   624Ôćĺ    let system_prompt = build_agent_system_prompt(&catalog, context);
   625Ôćĺ   625Ôćĺ
   626Ôćĺ   626Ôćĺ    // Initial state: [system, user(goal)].
   627Ôćĺ   627Ôćĺ    let mut agent_state = AgentState {
   628Ôćĺ   628Ôćĺ        iteration: 0,
   629Ôćĺ   629Ôćĺ        messages: vec![
   630Ôćĺ   630Ôćĺ            AgentMessage::system(system_prompt),
   631Ôćĺ   631Ôćĺ            AgentMessage::user(goal.to_string()),
   632Ôćĺ   632Ôćĺ        ],
   633Ôćĺ   633Ôćĺ        tool_results: Vec::new(),
   634Ôćĺ   634Ôćĺ        completed: false,
   635Ôćĺ   635Ôćĺ    };
   636Ôćĺ   636Ôćĺ
   637Ôćĺ   637Ôćĺ    let mut all_tool_calls: Vec<ToolCall> = Vec::new();
   638Ôćĺ   638Ôćĺ    let mut final_text = String::new();
   639Ôćĺ   639Ôćĺ
   640Ôćĺ   640Ôćĺ    for i in 0..cap {
   641Ôćĺ   641Ôćĺ        agent_state.iteration = i + 1;
   642Ôćĺ   642Ôćĺ
   643Ôćĺ   643Ôćĺ        // ---- 2. Think: call the LLM with the current message stream ----
   644Ôćĺ   644Ôćĺ        let response_text = call_llm(&state, &agent_state.messages).await;
   645Ôćĺ   645Ôćĺ
   646Ôćĺ   646Ôćĺ        // ---- 3a. Check for explicit completion sentinel ----
   647Ôćĺ   647Ôćĺ        if response_text.contains(TASK_COMPLETE_SENTINEL) {
   648Ôćĺ   648Ôćĺ            final_text = response_text;
   649Ôćĺ   649Ôćĺ            agent_state.completed = true;
   650Ôćĺ   650Ôćĺ            break;
   651Ôćĺ   651Ôćĺ        }
   652Ôćĺ   652Ôćĺ
   653Ôćĺ   653Ôćĺ        // ---- 3b. Parse tool calls from the response ----
   654Ôćĺ   654Ôćĺ        let calls = parse_tool_calls(&response_text);
   655Ôćĺ   655Ôćĺ        all_tool_calls.extend(calls.clone());
   656Ôćĺ   656Ôćĺ
   657Ôćĺ   657Ôćĺ        // ---- 3c. No tool calls + no sentinel Ôćĺ stop (incomplete) ----
   658Ôćĺ   658Ôćĺ        if calls.is_empty() {
   659Ôćĺ   659Ôćĺ            final_text = response_text;
   660Ôćĺ   660Ôćĺ            // `completed` stays false ÔÇö the agent stopped without signaling.
   661Ôćĺ   661Ôćĺ            break;
   662Ôćĺ   662Ôćĺ        }
   663Ôćĺ   663Ôćĺ
   664Ôćĺ   664Ôćĺ        // Record the assistant message that contained the tool calls, so
   665Ôćĺ   665Ôćĺ        // the next think step sees the full conversation context.
   666Ôćĺ   666Ôćĺ        agent_state.messages.push(AgentMessage::assistant(response_text.clone()));
   667Ôćĺ   667Ôćĺ
   668Ôćĺ   668Ôćĺ        // ---- 4. Execute: dispatch each tool call through the registry ----
   669Ôćĺ   669Ôćĺ        for call in &calls {
   670Ôćĺ   670Ôćĺ            let (output, success) = match Tool::from_request(&call.name, &call.params) {
   671Ôćĺ   671Ôćĺ                Some(tool) => registry.execute(&tool).await,
   672Ôćĺ   672Ôćĺ                None => (
   673Ôćĺ   673Ôćĺ                    format!(
   674Ôćĺ   674Ôćĺ                        "tool_not_found: no tool named \"{}\" in registry. \
   675Ôćĺ   675Ôćĺ                         Available tools: {}",
   676Ôćĺ   676Ôćĺ                        call.name,
   677Ôćĺ   677Ôćĺ                        catalog
   678Ôćĺ   678Ôćĺ                            .iter()
   679Ôćĺ   679Ôćĺ                            .map(|t| t.name)
   680Ôćĺ   680Ôćĺ                            .collect::<Vec<_>>()
   681Ôćĺ   681Ôćĺ                            .join(", ")
   682Ôćĺ   682Ôćĺ                    ),
   683Ôćĺ   683Ôćĺ                    false,
   684Ôćĺ   684Ôćĺ                ),
   685Ôćĺ   685Ôćĺ            };
   686Ôćĺ   686Ôćĺ
   687Ôćĺ   687Ôćĺ            let result = ToolResult {
   688Ôćĺ   688Ôćĺ                tool: call.name.clone(),
   689Ôćĺ   689Ôćĺ                output: output.clone(),
   690Ôćĺ   690Ôćĺ                success,
   691Ôćĺ   691Ôćĺ            };
   692Ôćĺ   692Ôćĺ            agent_state.tool_results.push(result);
   693Ôćĺ   693Ôćĺ
   694Ôćĺ   694Ôćĺ            // ---- 5. Observe: feed the tool result back into the stream ----
   695Ôćĺ   695Ôćĺ            agent_state
   696Ôćĺ   696Ôćĺ                .messages
   697Ôćĺ   697Ôćĺ                .push(AgentMessage::tool(call.name.clone(), output));
   698Ôćĺ   698Ôćĺ        }
   699Ôćĺ   699Ôćĺ
   700Ôćĺ   700Ôćĺ        // The "final text" is the most recent assistant response, so if
   701Ôćĺ   701Ôćĺ        // the next iteration hits the iteration cap we still return
   702Ôćĺ   702Ôćĺ        // something useful.
   703Ôćĺ   703Ôćĺ        final_text = response_text;
   704Ôćĺ   704Ôćĺ
   705Ôćĺ   705Ôćĺ        // ---- 6. Repeat (loop continues to the next think step) ----
   706Ôćĺ   706Ôćĺ    }
   707Ôćĺ   707Ôćĺ
   708Ôćĺ   708Ôćĺ    AgentRunResult {
   709Ôćĺ   709Ôćĺ        result: final_text,
   710Ôćĺ   710Ôćĺ        iterations: agent_state.iteration,
   711Ôćĺ   711Ôćĺ        tool_calls: all_tool_calls,
   712Ôćĺ   712Ôćĺ        completed: agent_state.completed,
   713Ôćĺ   713Ôćĺ    }
   714Ôćĺ   714Ôćĺ}
